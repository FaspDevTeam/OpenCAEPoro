50
51 if (val >= data[j][bId]) {
52 for (USI i = bId + 1; i < nRow; i++) {
53 if (val < data[j][i]) {
54 bId = i - 1;
55 for (USI k = 0; k < nCol; k++) {
56 slope[k] = (data[k][bId + 1] - data[k][bId]) /

58 outdata[k] = data[k][bId] + slope[k] * (val - data[j][bId]);

64 slope[k] = 0;
