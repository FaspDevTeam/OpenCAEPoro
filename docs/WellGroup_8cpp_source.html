18void WellGroup::InputParam(const ParamWell& paramWell)
19{
20 OCP_FUNCNAME;
21
22 USI len = paramWell.solSet.size();
23 solvents.resize(len);
24 for (USI i = 0; i < len; i++) {
25 solvents[i] = paramWell.solSet[i];
26 }
27
28 numWell = paramWell.well.size();
29 wellGroup.resize(numWell);
30 USI t = paramWell.criticalTime.size();
31 vector<USI> wellCriticalTime;
32 for (USI w = 0; w < numWell; w++) {
33 wellGroup[w].name = paramWell.well[w].name;
34 wellGroup[w].depth = paramWell.well[w].depth;
35 wellGroup[w].I = paramWell.well[w].I - 1;
36 wellGroup[w].J = paramWell.well[w].J - 1;
37
38 wellGroup[w].InputPerfo(paramWell.well[w]);
39
40 // opt
41 wellGroup[w].optSet.resize(t);
42 USI n = paramWell.well[w].optParam.size();
43 wellCriticalTime.clear();
44 wellCriticalTime.resize(n + 1);
45 for (USI i = 0; i < n; i++) {
46 wellCriticalTime[i] = paramWell.well[w].optParam[i].d;
47 }
48 wellCriticalTime.back() = t;
49 for (USI i = 0; i < n; i++) {
50 for (USI d = wellCriticalTime[i]; d < wellCriticalTime[i + 1]; d++) {
51 wellGroup[w].optSet[d] = WellOpt(paramWell.well[w].optParam[i].opt);
52 }
53 }
54 }
55}

149void WellGroup::CalIPRT(const Bulk& myBulk, OCP_DBL dt)
150{
151 OCP_FUNCNAME;
152
153 FGIR = 0;
154 FWIR = 0;
155 FOPR = 0;
156 FGPR = 0;
157 FWPR = 0;
158 for (USI w = 0; w < numWell; w++) {
159 wellGroup[w].WGIR = 0;
160 wellGroup[w].WWIR = 0;
161 wellGroup[w].WOPR = 0;
162 wellGroup[w].WGPR = 0;
163 wellGroup[w].WWPR = 0;
164
165 if (wellGroup[w].WellState()) {
166 if (wellGroup[w].WellType() == PROD) {
167 wellGroup[w].CalProdQiBO(myBulk, dt);
168 } else {
169 wellGroup[w].CalInjQiBO(myBulk, dt);
170 }
171 }
172 FGIR += wellGroup[w].WGIR;
173 FWIR += wellGroup[w].WWIR;
174 FOPR += wellGroup[w].WOPR;
175 FGPR += wellGroup[w].WGPR;
176 FWPR += wellGroup[w].WWPR;
177 }
178 FGIT += FGIR * dt;
179 FWIT += FWIR * dt;
180 FOPT += FOPR * dt;
181 FGPt += FGPR * dt;
182 FWPT += FWPR * dt;
183}

206OCP_INT WellGroup::CheckP(const Bulk& myBulk)
207{
208 OCP_FUNCNAME;
209
210 // 0 : All correct
211 // 1 : negative P, cut the timestep and resolve
212 // 2.1 : change well mode to BHP, resolve
213 // 2.2 : crossflow happens, then close corresponding perf, resolve
214 bool flag2 = false;
215 bool flag3 = false;
216
217 for (USI w = 0; w < numWell; w++) {
218 if (wellGroup[w].WellState()) {
219
220 OCP_INT flag = wellGroup[w].CheckP(myBulk);
221#ifdef _DEBUG
222 // wellGroup[w].ShowPerfStatus();
223#endif // _DEBUG
224 switch (flag) {
225 case 1:
226 return 1;
227 case 2:
228 flag2 = true;
229 break;
230 case 3:
231 flag3 = true;
232 break;
233 default:
234 break;
235 }
236 }
237 }
238
239 if (flag2 || flag3) return 2;
240
241 return 0;
242}
