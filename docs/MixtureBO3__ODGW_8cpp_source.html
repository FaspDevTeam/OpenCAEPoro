31void BOMixture_ODGW::InitFlash(const OCP_DBL& Pin, const OCP_DBL& Pbbin, const OCP_DBL& Tin,
32 const OCP_DBL* Sjin, const OCP_DBL& Vpore,
33 const OCP_DBL* Ziin)
34{
35 for (USI j = 0; j < 3; j++) phaseExist[j] = false;
36 for (USI i = 0; i < 3; i++) Ni[i] = 0;
37 fill(xij.begin(), xij.end(), 0.0);
38
39
40 P = Pin;
41 S[1] = Sjin[1]; // gas saturation
42 S[2] = Sjin[2]; // water saturation
43
44 // Water Property
45 PVTW.Eval_All(0, P, data, cdata);
46 OCP_DBL Pw0 = data[0];
47 OCP_DBL bw0 = data[1];
48 OCP_DBL cbw = data[2];
49 OCP_DBL bw = bw0 * (1 - cbw * (P - Pw0));
50 OCP_DBL bwp = -cbw * bw0;
51
52 mu[2] = data[3];
53 rho[2] = std_RhoW / bw;
54 xi[2] = 1 / (CONV1 * bw);
55 Ni[2] = Vpore * S[2] * xi[2];
56
57 USI phasecae;
58
59 if (1 - S[1] - S[2] < TINY) {
60 if (S[1] < TINY)
61 phasecae = PHASE_W; // case 1 : water, no oil, no gas
62 else
63 phasecae = PHASE_GW; // case 2 : water, gas, no oil
64 } else if (S[1] < TINY)
65 phasecae = PHASE_OW; // case 3 : water, oil, no gas
66 else
67 phasecae = PHASE_ODGW; // case 4 : water, oil, gas
68
69 switch (phasecae) {
70 case PHASE_W: {
71 // water
72 phaseExist[2] = true;
73 S[0] = 0;
74 S[1] = 0;
75 S[2] = 1;
76 xij[2 * 3 + 2] = 1;
77
78 // hypothetical Oil property
79 PVCO.Eval_All(0, P, data, cdata);
80 OCP_DBL rs = data[1];
81 OCP_DBL bo = data[2];
82
83 // hypothetical Gas property
84 PVDG.Eval_All(0, P, data, cdata);
85 OCP_DBL bg = data[1] * (CONV1 / 1000);
86
87 // total
88 v[0] = 0;
89 v[1] = 0;
90 v[2] = CONV1 * Ni[2] * bw;
91 vf = v[2];
92 vfp = CONV1 * Ni[2] * bwp;
93 vfi[0] = CONV1 * bo - 1000 * rs * bg;
94 vfi[1] = 1000 * bg;
95 vfi[2] = CONV1 * bw;
96
97 break;
98 }
99 case PHASE_GW: {
100 // water, gas
101 phaseExist[1] = true;
102 phaseExist[2] = true;
103 xij[1 * 3 + 1] = 1;
104 xij[2 * 3 + 2] = 1;
105
106 // hypothetical Oil property
107 PVCO.Eval_All(0, P, data, cdata);
108 OCP_DBL rs = data[1];
109 OCP_DBL bo = data[2];
110
111 // gas property
112 PVDG.Eval_All(0, P, data, cdata);
113 OCP_DBL bg = data[1] * (CONV1 / 1000);
114 OCP_DBL cbg = cdata[1] * (CONV1 / 1000);
115
116 mu[1] = data[2];
117 xi[1] = 1 / 1000 / bg;
118 rho[1] = std_RhoG / bg;
119 Ni[1] = Vpore * S[1] * xi[1];
120
121 v[0] = 0;
122 v[1] = 1000 * Ni[1] * bg; // Ni[0] = 0;
123 v[2] = CONV1 * Ni[2] * bw;
124 // total
125 vf = v[1] + v[2];
126 S[0] = 0;
127 S[1] = v[1] / vf;
128 S[2] = v[2] / vf;
129 vfp = 1000 * Ni[1] * cbg + CONV1 * Ni[2] * bwp;
130 vfi[0] = CONV1 * bo - 1000 * rs * bg;
131 vfi[1] = 1000 * bg;
132 vfi[2] = CONV1 * bw;
133
134 break;
135 }
136 case PHASE_OW: {
137 // water, oil, unsaturated
138 phaseExist[0] = true;
139 phaseExist[2] = true;
140
141 // oil property
142 OCP_DBL Pbb = Pbbin;
143 PVCO.Eval_All(0, Pbb, data, cdata);
144 OCP_DBL rs = data[1];
145 OCP_DBL bosat = data[2];
146 OCP_DBL muosat = data[3];
147 OCP_DBL cbosat = data[4];
148 OCP_DBL cmuosat = data[5];
149 OCP_DBL bo = bosat * (1 - cbosat * (P - Pbb));
150 OCP_DBL bop = -bosat * cbosat;
151 OCP_DBL dBo_drs = bo / bosat * cdata[2] +
152 bosat * (cdata[4] * (Pbb - P) + cbosat * cdata[0]);
153 dBo_drs /= cdata[1];
154
155 Ni[0] = Vpore * (1 - S[1] - S[2]) / (CONV1 * bo);
156 Ni[1] = Ni[0] * rs;
157 xi[0] = (1 + rs) / (CONV1 * bo);
158 rho[0] = (std_RhoO + (1000 / CONV1) * rs * std_RhoG) / bo;
159 mu[0] = muosat * (1 + cmuosat * (P - Pbb));
160
161 xij[0 * 3 + 0] = Ni[0] / (Ni[0] + Ni[1]);
162 xij[0 * 3 + 1] = 1 - xij[0 * 3 + 0];
163 xij[1 * 3 + 1] = 1;
164 xij[2 * 3 + 2] = 1;
165
166 // total
167 v[0] = CONV1 * Ni[0] * bo;
168 v[2] = CONV1 * Ni[2] * bw;
169 vf = v[0] + v[2];
170 S[0] = v[0] / vf;
171 S[1] = 0;
172 S[2] = v[2] / vf;
173 vfp = CONV1 * (Ni[0] * bop + Ni[2] * bwp);
174 vfi[0] = CONV1 * (bo - dBo_drs * (Ni[1] / Ni[0]));
175 vfi[1] = CONV1 * dBo_drs;
176 vfi[2] = CONV1 * bw;
177
178 break;
179 }
180 case PHASE_ODGW: {
181 phaseExist[0] = true;
182 phaseExist[1] = true;
183 phaseExist[2] = true;
184
185 // oil property
186 PVCO.Eval_All(0, P, data, cdata);
187 OCP_DBL rs = data[1];
188 OCP_DBL bo = data[2];
189 OCP_DBL crs = cdata[1];
190 OCP_DBL cbosat = cdata[2];
191
192 mu[0] = data[3];
193 Ni[0] = Vpore * (1 - S[1] - S[2]) / (CONV1 * bo);
194 xi[0] = (1 + rs) / (CONV1 * bo);
195 rho[0] = (std_RhoO + (1000 / CONV1) * rs * std_RhoG) / bo;
196
197 // gas property
198 PVDG.Eval_All(0, P, data, cdata);
199 OCP_DBL bg = data[1] * (CONV1 / 1000);
200 OCP_DBL cbg = cdata[1] * (CONV1 / 1000);
201 Ni[1] = Vpore * S[1] / bg / 1000 + Ni[0] * rs;
202 xi[1] = 1 / CONV1 / data[1];
203 rho[1] = std_RhoG / bg;
204 mu[1] = data[2];
205
206 xij[0 * 3 + 0] = 1 / (1 + rs);
207 xij[0 * 3 + 1] = 1 - xij[0 * 3 + 0];
208 xij[1 * 3 + 1] = 1;
209 xij[2 * 3 + 2] = 1;
210
211 // total
212 v[0] = CONV1 * Ni[0] * bo;
213 v[1] = 1000 * (Ni[1] - rs * Ni[0]) * bg;
214 v[2] = CONV1 * Ni[2] * bw;
215
216 vf = v[0] + v[1] + v[2];
217 S[0] = v[0] / vf;
218 S[1] = v[1] / vf;
219 S[2] = v[2] / vf;
220 vfp = CONV1 * Ni[0] * cbosat +
221 1000 * (-crs * Ni[0] * bg + (Ni[1] - rs * Ni[0]) * cbg) +
222 CONV1 * Ni[2] * bwp;
223 vfi[0] = CONV1 * bo - 1000 * rs * bg;
224 vfi[1] = 1000 * bg;
225 vfi[2] = CONV1 * bw;
226
227 break;
228 }
229 }

433void BOMixture_ODGW::FlashDeriv(const OCP_DBL& Pin, const OCP_DBL& Tin,
434 const OCP_DBL* Niin, const USI& ftype, const USI& lastNP,
435 const OCP_DBL* lastKs)
436{
437 for (USI j = 0; j < 3; j++) {
438 phaseExist[j] = false;
439 rhoP[j] = 0; xiP[j] = 0; muP[j] = 0;
440 }
441 fill(xij.begin(), xij.end(), 0.0);
442 fill(rhox.begin(), rhox.end(), 0.0);
443 fill(xix.begin(), xix.end(), 0.0);
444 fill(mux.begin(), mux.end(), 0.0);
445 fill(dXsdXp.begin(), dXsdXp.end(), 0.0);
446
447
448 P = Pin;
449 Nt = 0;
450 for (USI i = 0; i < numCom; i++) {
451 Ni[i] = Niin[i];
452 Nt += Ni[i];
453 }
454
455 // Water property
456 PVTW.Eval_All(0, P, data, cdata);
457 OCP_DBL Pw0 = data[0];
458 OCP_DBL bw0 = data[1];
459 OCP_DBL cbw = data[2];
460 OCP_DBL bw = bw0 * (1 - cbw * (P - Pw0));
461 OCP_DBL bwp = -cbw * bw0;
462
463 mu[2] = data[3];
464 xi[2] = 1 / CONV1 / bw;
465 rho[2] = std_RhoW / bw;
466
467 muP[2] = cdata[3];
468 xiP[2] = -bwp / (bw * bw * CONV1);
469 rhoP[2] = CONV1 * xiP[2] * std_RhoW;
470
471 USI phasecase;
472 OCP_DBL Rs_sat = PVCO.Eval(0, P, 1);
473
474 if (Ni[0] < Nt * TINY) {
475 if (Ni[1] <= Ni[0] * Rs_sat)
476 phasecase = PHASE_W; // water, no oil, no gas
477 else
478 phasecase = PHASE_GW; // water, gas, no oil
479 }
480 // Beacareful when NO = NG = 0, if it's possible.
481 else if (Ni[1] <= Ni[0] * Rs_sat)
482 phasecase = PHASE_OW; // water, oil, no gas
483 else
484 phasecase = PHASE_ODGW; // water, oil ,gas
485
486 switch (phasecase) {
487 case PHASE_W: {
488 // water
489 phaseExist[2] = true;
490 S[0] = 0;
491 S[1] = 0;
492 S[2] = 1;
493 xij[2 * 3 + 2] = 1;
494
495 // hypothetical Oil property
496 PVCO.Eval_All(0, P, data, cdata);
497 OCP_DBL rs = data[1];
498 OCP_DBL bo = data[2];
499
500 // hypothetical Gas property
501 PVDG.Eval_All(0, P, data, cdata);
502 OCP_DBL bg = data[1] * (CONV1 / 1000);
503
504 // total
505 v[0] = 0;
506 v[1] = 0;
507 v[2] = CONV1 * Ni[2] * bw;
508 vf = v[2];
509 vfp = CONV1 * Ni[2] * bwp;
510 vfi[0] = CONV1 * bo - 1000 * rs * bg;
511 vfi[1] = 1000 * bg;
512 vfi[2] = CONV1 * bw;
513
514 // dXsdXp[0] = 0; // dSo / dP
515 dXsdXp[1] = CONV1 * bo / vf; // dSo / dNo
516 // dXsdXp[2] = 0; // dSo / dNg;
517 // dXsdXp[3] = 0; // dSo / dNw;
518
519 // dXsdXp[1 * 4 + 0] = 0; // dSg / dP
520 // dXsdXp[1 * 4 + 1] = 0; // dSg / dNo
521 dXsdXp[1 * 4 + 2] = 1000 * bg / vf; // dSg / dNg
522 // dXsdXp[1 * 4 + 3] = 0; // dSg / dNw
523
524 dXsdXp[2 * 4 + 0] = (Ni[2] * bwp * CONV1 - S[2] * vfp) / vf; // dSw / dP
525 dXsdXp[2 * 4 + 1] = -S[2] * vfi[0] / vf; // dSw / dNo
526 dXsdXp[2 * 4 + 2] = -S[2] * vfi[1] / vf; // dSw / dNg
527 dXsdXp[2 * 4 + 3] = (CONV1 * bw - S[2] * vfi[2]) / vf; // dSw / dNw
528
529 break;
530 }
531 case PHASE_GW: {
532 // water, gas
533 phaseExist[1] = true;
534 phaseExist[2] = true;
535 xij[1 * 3 + 1] = 1;
536 xij[2 * 3 + 2] = 1;
537
538 // hypothetical Oil property
539 PVCO.Eval_All(0, P, data, cdata);
540 OCP_DBL rs = data[1];
541 OCP_DBL bo = data[2];
542
543 // gas property
544 PVDG.Eval_All(0, P, data, cdata);
545 OCP_DBL bg = data[1] * (CONV1 / 1000);
546 OCP_DBL cbg = cdata[1] * (CONV1 / 1000);
547
548 mu[1] = data[2];
549 xi[1] = 1 / 1000 / bg;
550 rho[1] = std_RhoG / bg;
551
552 muP[1] = cdata[2];
553 xiP[1] = -cdata[1] / (CONV1 * data[1] * data[1]);
554 rhoP[1] = 1000 * std_RhoG * xiP[1];
555
556 // total
557 v[0] = 0;
558 v[1] = 1000 * (Ni[1] - rs * Ni[0]) * bg;
559 v[2] = CONV1 * Ni[2] * bw;
560
561#ifdef _DEBUG
562 if (v[1] <= 0) {
563 OCP_ABORT("gas volume <= 0");
564 }
565#endif // _DEBUG
566
567 vf = v[1] + v[2];
568 S[0] = 0;
569 S[1] = v[1] / vf;
570 S[2] = v[2] / vf;
571 vfp = 1000 * Ni[1] * cbg + CONV1 * Ni[2] * bwp;
572 vfi[0] = CONV1 * bo - 1000 * rs * bg;
573 vfi[1] = 1000 * bg;
574 vfi[2] = CONV1 * bw;
575
576 // dXsdXp[0] = 0; // dSo / dP
577 dXsdXp[1] = CONV1 * bo / vf; // dSo / dNo
578 // dXsdXp[2] = 0; // dSo / dNg
579 // dXsdXp[3] = 0; // dSo / dNw
580
581 dXsdXp[1 * 4 + 0] = (1000 * Ni[1] * cbg - S[1] * vfp) / vf; // dSg / dP
582 dXsdXp[1 * 4 + 1] = (-1000 * rs * bg - S[1] * vfi[0]) / vf; // dSg / dNo
583 dXsdXp[1 * 4 + 2] = (1000 * bg - S[1] * vfi[1]) / vf; // dSg / dNg
584 dXsdXp[1 * 4 + 3] = -S[1] * vfi[2] / vf; // dSg / dNw
585
586 dXsdXp[2 * 4 + 0] = (CONV1 * Ni[2] * bwp - S[2] * vfp) / vf; // dSw / dP
587 dXsdXp[2 * 4 + 1] = -S[2] * vfi[0] / vf; // dSw / dNo
588 dXsdXp[2 * 4 + 2] = -S[2] * vfi[1] / vf; // dSw / dNg
589 dXsdXp[2 * 4 + 3] = (CONV1 * bw - S[2] * vfi[2]) / vf; // dSw / dNw
590
591 break;
592 }
593 case PHASE_OW: {
594 // water, oil
595 phaseExist[0] = true;
596 phaseExist[2] = true;
597 xij[0 * 3 + 0] = Ni[0] / (Ni[0] + Ni[1]);
598 xij[0 * 3 + 1] = 1 - xij[0 * 3 + 0];
599 xij[1 * 3 + 1] = 1;
600 xij[2 * 3 + 2] = 1;
601
602 // oil property
603 OCP_DBL rs = Ni[1] / Ni[0];
604 PVCO.Eval_All(1, rs, data, cdata);
605 OCP_DBL pbb = data[0];
606 OCP_DBL bosat = data[2];
607 OCP_DBL muosat = data[3];
608 OCP_DBL cbosat = data[4];
609 OCP_DBL cmuosat = data[5];
610 OCP_DBL bo = bosat * (1 - cbosat * (P - pbb));
611 OCP_DBL bop = -bosat * cbosat;
612 OCP_DBL dBo_drs = (1 - cbosat * (P - pbb)) * cdata[2] +
613 bosat * (cdata[4] * (pbb - P) + cbosat * cdata[0]);
614
615 mu[0] = muosat * (1 + cmuosat * (P - pbb));
616 xi[0] = (1 + rs) / (CONV1 * bo);
617 rho[0] = (std_RhoO + (1000 / CONV1) * rs * std_RhoG) / bo;
618
619 muP[0] = muosat * cmuosat;
620 xiP[0] = -(1 + rs) * bop / (CONV1 * bo * bo);
621 rhoP[0] = -(std_RhoO + (1000/CONV1) * rs * std_RhoG) / (bo * bo) * bop;
622
623 OCP_DBL muo_rs = mu[0] / muosat * cdata[3] + muosat * (cdata[5] * (P - pbb) - cmuosat * cdata[0]);
624 OCP_DBL xio_rs = 1 / CONV1 / bo - (1 + rs) * dBo_drs / (CONV1 * bo * bo);
625 OCP_DBL rhoo_rs = (1000 / CONV1) * std_RhoG / bo - (std_RhoO + (1000 / CONV1) * rs * std_RhoG) * dBo_drs / (bo * bo);
626
627 // total
628 v[0] = CONV1 * Ni[0] * bo;
629 v[2] = CONV1 * Ni[2] * bw;
630 vf = v[0] + v[2];
631 S[0] = v[0] / vf;
632 S[1] = 0;
633 S[2] = v[2] / vf;
634 vfp = CONV1 * (Ni[0] * bop + Ni[2] * bwp);
635 vfi[0] = CONV1 * (bo - dBo_drs * (Ni[1] / Ni[0]));
636 vfi[1] = CONV1 * dBo_drs;
637 vfi[2] = CONV1 * bw;
638
639
640 dXsdXp[0] = (-CONV1 * Ni[0] * cbosat * bosat - S[0] * vfp) / vf; // dSo / dP
641 dXsdXp[1] = (CONV1 * bo - CONV1 * dBo_drs * (Ni[1] / Ni[0]) - S[0] * vfi[0]) / vf; // dSg / dNo
642 dXsdXp[2] = (CONV1 * dBo_drs - S[0] * vfi[1]) / vf; // dSg / dNg
643 dXsdXp[3] = -S[0] / vf * vfi[2]; // dSg / dNw
644
645 dXsdXp[2 * 4 + 0] = (Ni[2] * bwp * CONV1 - S[2] * vfp) / vf; // dSw / dP
646 dXsdXp[2 * 4 + 1] = -S[2] * vfi[0] / vf; // dSw / dNo
647 dXsdXp[2 * 4 + 2] = -S[2] * vfi[1] / vf; // dSw / dNg
648 dXsdXp[2 * 4 + 3] = (CONV1 * bw - S[2] * vfi[2]) / vf; // dSw / dNw
649
650 dXsdXp[3 * 4 + 1] = Ni[1] / pow((Ni[0] + Ni[1]), 2); // d Xoo / d No
651 dXsdXp[3 * 4 + 2] = -Ni[0] / pow((Ni[0] + Ni[1]), 2); // d Xoo / d Ng
652 dXsdXp[4 * 4 + 1] = -dXsdXp[3 * 4 + 1]; // d Xgo / d No
653 dXsdXp[4 * 4 + 2] = -dXsdXp[3 * 4 + 2]; // d Xgo / d Ng
654
655 OCP_DBL tmp = xij[0] * xij[0];
656
657 mux[0] = -muo_rs * xij[1] / tmp; // dMuo / dXoo
658 mux[1] = muo_rs / xij[0]; // dMuo / dXgo
659
660 xix[0] = -xio_rs * xij[1] / tmp; // dXio / dXoo
661 xix[1] = xio_rs / xij[0]; // dXio / dXgo
662
663 rhox[0] = -rhoo_rs * xij[1] / tmp; // dRhoo / dXoo
664 rhox[1] = rhoo_rs / xij[0]; // dRhoo / dXgo
665
666 break;
667 }
668 case PHASE_ODGW: {
669 phaseExist[0] = true;
670 phaseExist[1] = true;
671 phaseExist[2] = true;
672
673 // oil property
674 PVCO.Eval_All(0, P, data, cdata);
675 OCP_DBL rs = data[1];
676 OCP_DBL bo = data[2];
677 OCP_DBL crs = cdata[1];
678 OCP_DBL cbosat = cdata[2];
679
680 mu[0] = data[3];
681 xi[0] = (1 + rs) / (CONV1 * bo);
682 rho[0] = (std_RhoO + (1000 / CONV1) * rs * std_RhoG) / bo;
683
684 muP[0] = cdata[3];
685 xiP[0] = (crs * bo - (1 + rs) * cbosat) / (bo * bo) / CONV1;
686 rhoP[0] = (1000 / CONV1) * std_RhoG * crs / bo - (std_RhoO + (1000/CONV1) * rs * std_RhoG) * cbosat / (bo * bo);
687
688 // gas property
689 PVDG.Eval_All(0, P, data, cdata);
690 OCP_DBL bg = data[1] * (CONV1 / 1000);
691 OCP_DBL cbg = cdata[1] * (CONV1 / 1000);
692
693 mu[1] = data[2];
694 xi[1] = 1 / CONV1 / data[1];
695 rho[1] = std_RhoG / bg;
696
697 muP[1] = cdata[2];
698 xiP[1] = -cdata[1] / (CONV1 * data[1] * data[1]);
699 rhoP[1] = 1000 * std_RhoG * xiP[1];
700
701 // total
702 xij[0 * 3 + 0] = 1 / (1 + rs);
703 xij[0 * 3 + 1] = 1 - xij[0 * 3 + 0];
704 xij[1 * 3 + 1] = 1;
705 xij[2 * 3 + 2] = 1;
706
707 v[0] = CONV1 * Ni[0] * bo;
708 v[1] = 1000 * (Ni[1] - rs * Ni[0]) * bg;
709 v[2] = CONV1 * Ni[2] * bw;
710 vf = v[0] + v[1] + v[2];
711 S[0] = v[0] / vf;
712 S[1] = v[1] / vf;
713 S[2] = v[2] / vf;
714 vfp = CONV1 * Ni[0] * cbosat +
715 1000 * (-crs * Ni[0] * bg + (Ni[1] - rs * Ni[0]) * cbg) +
716 CONV1 * Ni[2] * bwp;
717 vfi[0] = CONV1 * bo - 1000 * rs * bg;
718 vfi[1] = 1000 * bg;
719 vfi[2] = CONV1 * bw;
720
721
722 dXsdXp[0] = (CONV1 * Ni[0] * cbosat - S[0] * vfp)/ vf; // dSo / dP
723 dXsdXp[1] = (CONV1 * bo - S[0] * vfi[0]) / vf; // dSo / dNo
724 dXsdXp[2] = -S[0] * vfi[1] / vf; // dSo / dNg
725 dXsdXp[3] = -S[0] * vfi[2] / vf; // dSo / dNw
726
727 dXsdXp[1 * 4 + 0] = (1000 * (Ni[1] - rs * Ni[0]) * cbg - 1000 * Ni[0] * bg * crs - S[1] * vfp) / vf; // dSg / dP
728 dXsdXp[1 * 4 + 1] = (-1000 * rs * bg - S[1] * vfi[0]) / vf; // dSg / dNo
729 dXsdXp[1 * 4 + 2] = (1000 * bg - S[1] * vfi[1]) / vf; // dSg / dNg
730 dXsdXp[1 * 4 + 3] = -S[1] * vfi[2] / vf; // dSg / dNw
731
732 dXsdXp[2 * 4 + 0] = (Ni[2] * bwp * CONV1 - S[2] * vfp) / vf; // dSw / dP
733 dXsdXp[2 * 4 + 1] = -S[2] * vfi[0] / vf; // dSw / dNo
734 dXsdXp[2 * 4 + 2] = -S[2] * vfi[1] / vf; // dSw / dNg
735 dXsdXp[2 * 4 + 3] = (CONV1 * bw - S[2] * vfi[2]) / vf; // dSw / dNw
736
737 dXsdXp[3 * 4 + 0] = -crs / ((1 + rs) * (1 + rs)); // d Xoo / dP
738 dXsdXp[4 * 4 + 0] = -dXsdXp[3 * 4 + 0]; // d Xgo / dP
739
740 break;
741 }
742 }

746 if (!isfinite(v)) {
747 OCP_ABORT("Nan in dXsdXp!");
748 }
749 }
750#endif
751}
