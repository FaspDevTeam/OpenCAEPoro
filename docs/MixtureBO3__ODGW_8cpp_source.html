14void BOMixture::BOFlash_Sj_ODGW(const OCP_DBL& Pin, const OCP_DBL& Pbbin,
15 const OCP_DBL* Sjin, const OCP_DBL& Vpore)
16{
17
18 phaseExist.assign(numPhase, false);
19 xij.assign(numPhase * numCom, 0);
20 Ni.assign(numCom, 0);
21
22 P = Pin;
23 for (USI j = 0; j < numPhase; j++) S[j] = Sjin[j];
24
25 // Water Property
26 PVTW.Eval_All(0, P, data, cdata);
27 OCP_DBL Pw0 = data[0];
28 OCP_DBL bw0 = data[1];
29 OCP_DBL cbw = data[2];
30 OCP_DBL bw = bw0 * (1 - cbw * (P - Pw0));
31 OCP_DBL bwp = -cbw * bw0;
32
33 mu[2] = data[3];
34 rho[2] = std_RhoW / bw;
35 xi[2] = 1 / (CONV1 * bw);
36 Ni[2] = Vpore * S[2] * xi[2];
37
38 USI phasecae;
39
40 if (1 - S[1] - S[2] < TINY) {
41 if (S[1] < TINY)
42 phasecae = PHASE_W; // case 1 : water, no oil, no gas
43 else
44 phasecae = PHASE_GW; // case 2 : water, gas, no oil
45 } else if (S[1] < TINY)
46 phasecae = PHASE_OW; // case 3 : water, oil, no gas
47 else
48 phasecae = PHASE_ODGW; // case 4 : water, oil, gas
49
50 switch (phasecae) {
51 case PHASE_W: {
52 // water
53 phaseExist[2] = true;
54 S[0] = 0;
55 S[1] = 0;
56 S[2] = 1;
57 xij[2 * 3 + 2] = 1;
58
59 // hypothetical Oil property
60 PVCO.Eval_All(0, P, data, cdata);
61 OCP_DBL rs = data[1];
62 OCP_DBL bo = data[2];
63
64 // hypothetical Gas property
65 PVDG.Eval_All(0, P, data, cdata);
66 OCP_DBL bg = data[1] * (CONV1 / 1000);
67
68 // total
69 v[0] = 0;
70 v[1] = 0;
71 v[2] = CONV1 * Ni[2] * bw;
72 vf = v[2];
73 vfp = CONV1 * Ni[2] * bwp;
74 vfi[0] = CONV1 * bo - 1000 * rs * bg;
75 vfi[1] = 1000 * bg;
76 vfi[2] = CONV1 * bw;
77
78 break;
79 }
80 case PHASE_GW: {
81 // water, gas
82 phaseExist[1] = true;
83 phaseExist[2] = true;
84 xij[1 * 3 + 1] = 1;
85 xij[2 * 3 + 2] = 1;
86
87 // hypothetical Oil property
88 PVCO.Eval_All(0, P, data, cdata);
89 OCP_DBL rs = data[1];
90 OCP_DBL bo = data[2];
91
92 // gas property
93 PVDG.Eval_All(0, P, data, cdata);
94 OCP_DBL bg = data[1] * (CONV1 / 1000);
95 OCP_DBL cbg = cdata[1] * (CONV1 / 1000);
96
97 mu[1] = data[2];
98 xi[1] = 1 / bg / 1000;
99 rho[1] = std_RhoG / bg;
100 Ni[1] = Vpore * S[1] * xi[1];
101
102 v[0] = 0;
103 v[1] = 1000 * Ni[1] * bg; // Ni[0] = 0;
104 v[2] = CONV1 * Ni[2] * bw;
105 // total
106 vf = v[1] + v[2];
107 S[0] = 0;
108 S[1] = v[1] / vf;
109 S[2] = v[2] / vf;
110 vfp = 1000 * Ni[1] * cbg + CONV1 * Ni[2] * bwp;
111 vfi[0] = CONV1 * bo - 1000 * rs * bg;
112 vfi[1] = 1000 * bg;
113 vfi[2] = CONV1 * bw;
114
115 break;
116 }
117 case PHASE_OW: {
118 // water, oil, unsaturated
119 phaseExist[0] = true;
120 phaseExist[2] = true;
121
122 // oil property
123 OCP_DBL Pbb = Pbbin;
124 PVCO.Eval_All(0, Pbb, data, cdata);
125 OCP_DBL rs = data[1];
126 OCP_DBL bosat = data[2];
127 OCP_DBL muosat = data[3];
128 OCP_DBL cbosat = data[4];
129 OCP_DBL cmuosat = data[5];
130 OCP_DBL bo = bosat * (1 - cbosat * (P - Pbb));
131 OCP_DBL bop = -bosat * cbosat;
132 OCP_DBL dBo_drs = bo / bosat * cdata[2] +
133 bosat * (cdata[4] * (Pbb - P) + cbosat * cdata[0]);
134 dBo_drs /= cdata[1];
135
136 Ni[0] = Vpore * (1 - S[1] - S[2]) / (CONV1 * bo);
137 Ni[1] = Ni[0] * rs;
138 xi[0] = (1 + rs) / (CONV1 * bo);
139 rho[0] = (std_RhoO + (1000 / CONV1) * rs * std_RhoG) / bo;
140 mu[0] = muosat * (1 + cmuosat * (P - Pbb));
141
142 xij[0 * 3 + 0] = Ni[0] / (Ni[0] + Ni[1]);
143 xij[0 * 3 + 1] = 1 - xij[0 * 3 + 0];
144 xij[1 * 3 + 1] = 1;
145 xij[2 * 3 + 2] = 1;
146
147 // total
148 v[0] = CONV1 * Ni[0] * bo;
149 v[2] = CONV1 * Ni[2] * bw;
150 vf = v[0] + v[2];
151 S[0] = v[0] / vf;
152 S[1] = 0;
153 S[2] = v[2] / vf;
154 vfp = CONV1 * (Ni[0] * bop + Ni[2] * bwp);
155 vfi[0] = CONV1 * (bo - dBo_drs * (Ni[1] / Ni[0]));
156 vfi[1] = CONV1 * dBo_drs;
157 vfi[2] = CONV1 * bw;
158
159 break;
160 }
161 case PHASE_ODGW: {
162 phaseExist.assign(3, true);
163
164 // oil property
165 PVCO.Eval_All(0, P, data, cdata);
166 OCP_DBL rs = data[1];
167 OCP_DBL bo = data[2];
168 OCP_DBL crs = cdata[1];
169 OCP_DBL cbosat = cdata[2];
170
171 mu[0] = data[3];
172 Ni[0] = Vpore * (1 - S[1] - S[2]) / (CONV1 * bo);
173 xi[0] = (1 + rs) / bo / CONV1;
174 rho[0] = (std_RhoO + (1000 / CONV1) * rs * std_RhoG) / bo;
175
176 // gas property
177 PVDG.Eval_All(0, P, data, cdata);
178 OCP_DBL bg = data[1] * (CONV1 / 1000);
179 OCP_DBL cbg = cdata[1] * (CONV1 / 1000);
180 Ni[1] = Vpore * S[1] / bg / 1000 + Ni[0] * rs;
181 xi[1] = 1 / data[1] / CONV1;
182 rho[1] = std_RhoG / bg;
183 mu[1] = data[2];
184
185 xij[0 * 3 + 0] = 1 / (1 + rs);
186 xij[0 * 3 + 1] = 1 - xij[0 * 3 + 0];
187 xij[1 * 3 + 1] = 1;
188 xij[2 * 3 + 2] = 1;
189
190 // total
191 v[0] = CONV1 * Ni[0] * bo;
192 v[1] = 1000 * (Ni[1] - rs * Ni[0]) * bg;
193 v[2] = CONV1 * Ni[2] * bw;
194
195 vf = v[0] + v[1] + v[2];
196 S[0] = v[0] / vf;
197 S[1] = v[1] / vf;
198 S[2] = v[2] / vf;
199 vfp = CONV1 * Ni[0] * cbosat +
200 1000 * (-crs * Ni[0] * bg + (Ni[1] - rs * Ni[0]) * cbg) +
201 CONV1 * Ni[2] * bwp;
202 vfi[0] = CONV1 * bo - 1000 * rs * bg;
203 vfi[1] = 1000 * bg;
204 vfi[2] = CONV1 * bw;
205
206 break;
207 }
208 }

582 OCP_DBL cbosat = data[4];
583 OCP_DBL cmuosat = data[5];
584 OCP_DBL bo = bosat * (1 - cbosat * (P - pbb));
585 OCP_DBL bop = -bosat * cbosat;
586 OCP_DBL dBo_drs = (1 - cbosat * (P - pbb)) * cdata[2] +
587 bosat * (cdata[4] * (pbb - P) + cbosat * cdata[0]);
588
589 mu[0] = muosat * (1 + cmuosat * (P - pbb));
590 xi[0] = (1 + rs) / (CONV1 * bo);
591 rho[0] = (std_RhoO + (1000 / CONV1) * rs * std_RhoG) / bo;
592
593 muP[0] = muosat * cmuosat;
594 xiP[0] = -(1 + rs) * bop / (CONV1 * bo * bo);
595 rhoP[0] = -(std_RhoO + (1000 / CONV1) * rs * std_RhoG) / (bo * bo) * bop;
596
597 OCP_DBL muo_rs = mu[0] / muosat * cdata[3] +
598 muosat * (cdata[5] * (P - pbb) - cmuosat * cdata[0]);
599 OCP_DBL xio_rs = 1 / (CONV1 * bo) - (1 + rs) * dBo_drs / (CONV1 * bo * bo);
600 OCP_DBL rhoo_rs =
601 (1000 / CONV1) * std_RhoG / bo -
602 (std_RhoO + (1000 / CONV1) * rs * std_RhoG) * dBo_drs / (bo * bo);
603
604 // total
605 v[0] = CONV1 * Ni[0] * bo;
606 v[2] = CONV1 * Ni[2] * bw;
607 vf = v[0] + v[2];
608 S[0] = v[0] / vf;
609 S[1] = 0;
610 S[2] = v[2] / vf;
611 vfp = CONV1 * (Ni[0] * bop + Ni[2] * bwp);
612 vfi[0] = CONV1 * (bo - dBo_drs * (Ni[1] / Ni[0]));
613 vfi[1] = CONV1 * dBo_drs;
614 vfi[2] = CONV1 * bw;
615
616 dSec_dPri[0] =
617 (-CONV1 * Ni[0] * cbosat * bosat - S[0] * vfp) / vf; // dSo / dP
618 dSec_dPri[1] =
619 (CONV1 * bo - CONV1 * dBo_drs * (Ni[1] / Ni[0]) - S[0] * vfi[0]) /
620 vf; // dSg / dNo
621 dSec_dPri[2] = (CONV1 * dBo_drs - S[0] * vfi[1]) / vf; // dSg / dNg
622 dSec_dPri[3] = -S[0] / vf * vfi[2]; // dSg / dNw
623
624 dSec_dPri[2 * 4 + 0] = (Ni[2] * bwp * CONV1 - S[2] * vfp) / vf; // dSw / dP
625 dSec_dPri[2 * 4 + 1] = -S[2] * vfi[0] / vf; // dSw / dNo
626 dSec_dPri[2 * 4 + 2] = -S[2] * vfi[1] / vf; // dSw / dNg
627 dSec_dPri[2 * 4 + 3] = (CONV1 * bw - S[2] * vfi[2]) / vf; // dSw / dNw
628
629 dSec_dPri[3 * 4 + 1] = Ni[1] / pow((Ni[0] + Ni[1]), 2); // d Xoo / d No
630 dSec_dPri[3 * 4 + 2] = -Ni[0] / pow((Ni[0] + Ni[1]), 2); // d Xoo / d Ng
631 dSec_dPri[4 * 4 + 1] = -dSec_dPri[3 * 4 + 1]; // d Xgo / d No
632 dSec_dPri[4 * 4 + 2] = -dSec_dPri[3 * 4 + 2]; // d Xgo / d Ng
633
634 OCP_DBL tmp = xij[0] * xij[0];
635
636 mux[0] = -muo_rs * xij[1] / tmp; // dMuo / dXoo
637 mux[1] = muo_rs / xij[0]; // dMuo / dXgo
638
639 xix[0] = -xio_rs * xij[1] / tmp; // dXio / dXoo
640 xix[1] = xio_rs / xij[0]; // dXio / dXgo
641
642 rhox[0] = -rhoo_rs * xij[1] / tmp; // dRhoo / dXoo
643 rhox[1] = rhoo_rs / xij[0]; // dRhoo / dXgo
644
645 break;
646 }
647 case PHASE_ODGW: {
648 phaseExist.assign(3, true);
649
650 // oil property
651 PVCO.Eval_All(0, P, data, cdata);
652 OCP_DBL rs = data[1];
653 OCP_DBL bo = data[2];
654 OCP_DBL crs = cdata[1];
655 OCP_DBL cbosat = cdata[2];
656
657 mu[0] = data[3];
658 xi[0] = (1 + rs) / bo / CONV1;
659 rho[0] = (std_RhoO + (1000 / CONV1) * rs * std_RhoG) / bo;
660
661 muP[0] = cdata[3];
662 xiP[0] = crs / (CONV1 * bo) - (1 + rs) * cbosat / (CONV1 * bo * bo);
663 rhoP[0] = (1000 / CONV1) * std_RhoG * crs / bo -
664 (std_RhoO + (1000 / CONV1) * rs * std_RhoG) * cbosat / (bo * bo);
665
666 // gas property
667 PVDG.Eval_All(0, P, data, cdata);
668 OCP_DBL bg = data[1] * (CONV1 / 1000);
669 OCP_DBL cbg = cdata[1] * (CONV1 / 1000);
670
671 mu[1] = data[2];
672 xi[1] = 1 / CONV1 / data[1];
673 rho[1] = std_RhoG / bg;
674
675 muP[1] = cdata[2];
676 xiP[1] = -cdata[1] / (CONV1 * data[1] * data[1]);
677 rhoP[1] = 1000 * std_RhoG * xiP[1];
678
679 // total
680 xij[0 * 3 + 0] = 1 / (1 + rs);
681 xij[0 * 3 + 1] = 1 - xij[0 * 3 + 0];
682 xij[1 * 3 + 1] = 1;
683 xij[2 * 3 + 2] = 1;
684
685 v[0] = CONV1 * Ni[0] * bo;
686 v[1] = 1000 * (Ni[1] - rs * Ni[0]) * bg;
687 v[2] = CONV1 * Ni[2] * bw;
688 vf = v[0] + v[1] + v[2];
689 S[0] = v[0] / vf;
690 S[1] = v[1] / vf;
691 S[2] = v[2] / vf;
692 vfp = CONV1 * Ni[0] * cbosat +
693 1000 * (-crs * Ni[0] * bg + (Ni[1] - rs * Ni[0]) * cbg) +
694 CONV1 * Ni[2] * bwp;
695 vfi[0] = CONV1 * bo - 1000 * rs * bg;
696 vfi[1] = 1000 * bg;
697 vfi[2] = CONV1 * bw;
698
699 dSec_dPri[0] = (CONV1 * Ni[0] * cbosat - S[0] * vfp) / vf; // dSo / dP
700 dSec_dPri[1] = (CONV1 * bo - S[0] * vfi[0]) / vf; // dSo / dNo
701 dSec_dPri[2] = -S[0] * vfi[1] / vf; // dSo / dNg
702 dSec_dPri[3] = -S[0] * vfi[2] / vf; // dSo / dNw
703
704 dSec_dPri[1 * 4 + 0] = (1000 * (Ni[1] - rs * Ni[0]) * cbg -
705 1000 * Ni[0] * bg * crs - S[1] * vfp) /
706 vf; // dSg / dP
707 dSec_dPri[1 * 4 + 1] = (-1000 * rs * bg - S[1] * vfi[0]) / vf; // dSg / dNo
708 dSec_dPri[1 * 4 + 2] = (1000 * bg - S[1] * vfi[1]) / vf; // dSg / dNg
709 dSec_dPri[1 * 4 + 3] = -S[1] * vfi[2] / vf; // dSg / dNw
710
711 dSec_dPri[2 * 4 + 0] = (Ni[2] * bwp * CONV1 - S[2] * vfp) / vf; // dSw / dP
712 dSec_dPri[2 * 4 + 1] = -S[2] * vfi[0] / vf; // dSw / dNo
713 dSec_dPri[2 * 4 + 2] = -S[2] * vfi[1] / vf; // dSw / dNg
714 dSec_dPri[2 * 4 + 3] = (CONV1 * bw - S[2] * vfi[2]) / vf; // dSw / dNw
715
716 dSec_dPri[3 * 4 + 0] = -crs / ((1 + rs) * (1 + rs));
717 dSec_dPri[4 * 4 + 0] = -dSec_dPri[3 * 4 + 0];
718
719 break;
720 }
721 }

726 OCP_ABORT("Nan in dSec_dPri!");
727 }
