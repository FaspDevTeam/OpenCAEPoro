119void ScalarFaspSolver::AssembleMat(const vector<vector<USI>>& colId,
120 const vector<vector<OCP_DBL>>& val,
121 const OCP_USI& dim, const USI& blockDim,
122 vector<OCP_DBL>& rhs, vector<OCP_DBL>& u)
123{
124 // b & x
125 b.row = dim;
126 b.val = rhs.data();
127 x.row = dim;
128 x.val = u.data();
129 // A
130 OCP_USI nnz = 0;
131 for (OCP_USI i = 0; i < dim; i++) {
132 nnz += colId[i].size();
133 }
134
135 A.row = dim;
136 A.col = dim;
137 A.nnz = nnz;
138
139 // IA
140 OCP_USI count = 0;
141 A.IA[0] = 0;
142 for (OCP_USI i = 1; i < dim + 1; i++) {
143 USI nnz_Row = colId[i - 1].size();
144 A.IA[i] = A.IA[i - 1] + nnz_Row;
145
146 for (USI j = 0; j < nnz_Row; j++) {
147 A.JA[count] = colId[i - 1][j];
148 A.val[count] = val[i - 1][j];
149 count++;
150 }
151 }
152}

154OCP_INT ScalarFaspSolver::Solve(vector<OCP_DBL>& u)
155{
156 OCP_INT status = FASP_SUCCESS;
157
158 const OCP_INT print_level = inParam.print_level;
159 const OCP_INT solver_type = inParam.solver_type;
160 const OCP_INT precond_type = inParam.precond_type;
161 const OCP_INT output_type = inParam.output_type;
162
163 if (output_type) {
164 const char* outputfile = "out/Solver.out";
165 printf("Redirecting outputs to file: %s ...\n", outputfile);
166 freopen(outputfile, "w", stdout); // open a file for stdout
167 }
168
169 // Preconditioned Krylov methods
170 if (solver_type >= 1 && solver_type <= 20) {
171
172 // Using no preconditioner for Krylov iterative methods
173 if (precond_type == PREC_NULL) {
174 status = fasp_solver_dcsr_krylov(&A, &b, &x, &itParam);
175 }
176
177 // Using diag(A) as preconditioner for Krylov iterative methods
178 else if (precond_type == PREC_DIAG) {
179 status = fasp_solver_dcsr_krylov_diag(&A, &b, &x, &itParam);
180 }
181
182 // Using AMG as preconditioner for Krylov iterative methods
183 else if (precond_type == PREC_AMG || precond_type == PREC_FMG) {
184 if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
185 status = fasp_solver_dcsr_krylov_amg(&A, &b, &x, &itParam, &amgParam);
186 }
187
188 // Using ILU as preconditioner for Krylov iterative methods
189 else if (precond_type == PREC_ILU) {
190 if (print_level > PRINT_NONE) fasp_param_ilu_print(&iluParam);
191 status = fasp_solver_dcsr_krylov_ilu(&A, &b, &x, &itParam, &iluParam);
192 }
193
194 // Undefined iterative methods
195 else {
196 printf("### ERROR: Wrong preconditioner type %d!!!\n", precond_type);
197 status = ERROR_SOLVER_PRECTYPE;
198 }
199
200 }
201
202 // AMG as the iterative solver
203 else if (solver_type == SOLVER_AMG) {
204 if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
205 fasp_solver_amg(&A, &b, &x, &amgParam);
206 }
207
208 // Full AMG as the iterative solver
209 else if (solver_type == SOLVER_FMG) {
210 if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
211 fasp_solver_famg(&A, &b, &x, &amgParam);
212 }
213
214#if WITH_MUMPS // use MUMPS directly
215 else if (solver_type == SOLVER_MUMPS) {
216 status = fasp_solver_mumps(&A, &b, &x, print_level);
217 if (status >= 0) status = 1; // Direct solver returns 1
218 }
219#endif
220
221#if WITH_SuperLU // use SuperLU directly
222 else if (solver_type == SOLVER_SUPERLU) {
223 status = fasp_solver_superlu(&A, &b, &x, print_level);
224 if (status >= 0) status = 1; // Direct solver returns 1
225 }
226#endif
227
228#if WITH_UMFPACK // use UMFPACK directly
229 else if (solver_type == SOLVER_UMFPACK) {
230 // Need to sort the matrix A for UMFPACK to work
231 dCSRmat A_trans = fasp_dcsr_create(A.row, A.col, A.nnz);
232 fasp_dcsr_transz(&A, NULL, &A_trans);
233 fasp_dcsr_sort(&A_trans);
234 status = fasp_solver_umfpack(&A_trans, &b, &x, print_level);
235 fasp_dcsr_free(&A_trans);
236 if (status >= 0) status = 1; // Direct solver returns 1
237 }
238#endif
239
240#ifdef WITH_PARDISO // use PARDISO directly
241 else if (solver_type == SOLVER_PARDISO) {
242 fasp_dcsr_sort(&A);
243 status = fasp_solver_pardiso(&A, &b, &x, print_level);
244 if (status >= 0) status = 1; // Direct solver returns 1
245 }
246#endif
247
248 else {
249 printf("### ERROR: Wrong solver type %d!!!\n", solver_type);
250 status = ERROR_SOLVER_TYPE;
251 }
252
253 if (status < 0) {
254 printf("### ERROR: Solver failed! Exit status = %d.\n\n", status);
255 }
256
257 if (output_type) fclose(stdout);
258 return status;
259}

344void VectorFaspSolver::AssembleMat(const vector<vector<USI>>& colId,
345 const vector<vector<OCP_DBL>>& val,
346 const OCP_USI& dim, const USI& blockDim,
347 vector<OCP_DBL>& rhs, vector<OCP_DBL>& u)
348{
349 OCP_USI nrow = dim * blockDim;
350 // b & x
351 b.row = nrow;
352 b.val = rhs.data();
353 x.row = nrow;
354 x.val = u.data();
355 // fsc & order
356 fsc.row = nrow;
357 order.row = nrow;
358
359 // nnz
360 OCP_USI nnz = 0;
361 for (OCP_USI i = 0; i < dim; i++) {
362 nnz += colId[i].size();
363 }
364
365 // Asc
366 Asc.ROW = dim;
367 Asc.COL = dim;
368 Asc.nb = blockDim;
369 Asc.NNZ = nnz;
370
371 // A
372 A.ROW = dim;
373 A.COL = dim;
374 A.nb = blockDim;
375 A.NNZ = nnz;
376
377 OCP_USI count = 0;
378 OCP_USI count2 = 0;
379 OCP_USI size_row;
380 A.IA[0] = 0;
381 for (OCP_USI i = 1; i < dim + 1; i++) {
382 USI nnb_Row = colId[i - 1].size();
383 A.IA[i] = A.IA[i - 1] + nnb_Row;
384
385 for (USI j = 0; j < nnb_Row; j++) {
386 A.JA[count] = colId[i - 1][j];
387 count++;
388 }
389 size_row = nnb_Row * blockDim * blockDim;
390 for (USI k = 0; k < size_row; k++) {
391 A.val[count2] = val[i - 1][k];
392 count2++;
393 }
394 }
395}

397OCP_INT VectorFaspSolver::Solve(vector<OCP_DBL>& u)
398{
399 OCP_INT status = FASP_SUCCESS;
400
401 // Set local parameters
402 const OCP_INT print_level = inParam.print_level;
403 const OCP_INT solver_type = inParam.solver_type;
404 const OCP_INT decoup_type = inParam.decoup_type;
405 const OCP_INT precond_type = inParam.precond_type;
406 const OCP_INT output_type = inParam.output_type;
407
408 if (output_type) {
409 const char* outputfile = "../output/test.out";
410 printf("Redirecting outputs to file: %s ...\n", outputfile);
411 freopen(outputfile, "w", stdout); // open a file for stdout
412 }
413
414 fasp_dvec_set(x.row, &x, 0);
415
416 // Preconditioned Krylov methods
417 if (solver_type >= 1 && solver_type <= 10) {
418
419 // Preconditioned Krylov methods in BSR format
420 switch (precond_type) {
421 case PC_NULL:
422 status = fasp_solver_dbsr_krylov(&A, &b, &x, &itParam);
423 break;
424 case PC_DIAG:
425 status = fasp_solver_dbsr_krylov_diag(&A, &b, &x, &itParam);
426 break;
427 case PC_BILU:
428 status = fasp_solver_dbsr_krylov_ilu(&A, &b, &x, &itParam, &iluParam);
429 break;
430 case PC_FASP1:
431 Decoupling(&A, &b, &Asc, &fsc, &order, Dmat.data(), decoup_type);
432 status = fasp_solver_dbsr_krylov_FASP1a(
433 &Asc, &fsc, &x, &itParam, &iluParam, &amgParam, NULL, &order);
434 break;
435 case PC_FASP1_SHARE: // zhaoli 2021.03.24
436 Decoupling(&A, &b, &Asc, &fsc, &order, Dmat.data(), decoup_type);
437 status = fasp_solver_dbsr_krylov_FASP1a_share_interface(
438 &Asc, &fsc, &x, &itParam, &iluParam, &amgParam, NULL, &order,
439 RESET_CONST);
440 break;
441 case PC_FASP2:
442 Decoupling(&A, &b, &Asc, &fsc, &order, Dmat.data(), decoup_type);
443 status = fasp_solver_dbsr_krylov_FASP2(
444 &Asc, &fsc, &x, &itParam, &iluParam, &amgParam, NULL, &order);
445 break;
446 case PC_FASP3:
447 Decoupling(&A, &b, &Asc, &fsc, &order, Dmat.data(), decoup_type);
448 status = fasp_solver_dbsr_krylov_FASP3(
449 &Asc, &fsc, &x, &itParam, &iluParam, &amgParam, NULL, &order);
450 break;
451 case PC_FASP4_SHARE: // zhaoli 2021.04.24
452 Decoupling(&A, &b, &Asc, &fsc, &order, Dmat.data(), decoup_type);
453 status = fasp_solver_dbsr_krylov_FASP4_share_interface(
454 &Asc, &fsc, &x, &itParam, &iluParam, &amgParam, NULL, &order,
455 RESET_CONST);
456 break;
457 case PC_FASP5:
458 Decoupling(&A, &b, &Asc, &fsc, &order, Dmat.data(), decoup_type);
459 status = fasp_solver_dbsr_krylov_FASP5(
460 &Asc, &fsc, &x, &itParam, &iluParam, &amgParam, NULL, &order);
461 break;
462 default: // case PC_FASP4:
463 Decoupling(&A, &b, &Asc, &fsc, &order, Dmat.data(), decoup_type);
464 status = fasp_solver_dbsr_krylov_FASP4(
465 &Asc, &fsc, &x, &itParam, &iluParam, &amgParam, NULL, &order);
466 }
467 Dmat.assign(Dmat.size(), 0);
468 }
469
470#if WITH_MUMPS // use MUMPS directly
471 else if (solver_type == SOLVER_MUMPS) {
472 dCSRmat Acsr = fasp_format_dbsr_dcsr(&A);
473 status = fasp_solver_mumps(&Acsr, &b, &x, print_level);
474 fasp_dcsr_free(&Acsr);
475 if (status >= 0) status = 1; // Direct solver returns 1
476 }
477#endif
478
479#if WITH_SuperLU // use SuperLU directly
480 else if (solver_type == SOLVER_SUPERLU) {
481 dCSRmat Acsr = fasp_format_dbsr_dcsr(&A);
482 status = fasp_solver_superlu(&Acsr, &b, &x, print_level);
483 fasp_dcsr_free(&Acsr);
484 if (status >= 0) status = 1; // Direct solver returns 1
485 }
486#endif
487
488#if WITH_UMFPACK // use UMFPACK directly
489 else if (solver_type == SOLVER_UMFPACK) {
490 // Need to sort the matrix A for UMFPACK to work
491 dCSRmat Acsr = fasp_format_dbsr_dcsr(&A);
492 dCSRmat A_trans = fasp_dcsr_create(Acsr.row, Acsr.col, Acsr.nnz);
493 fasp_dcsr_transz(&Acsr, NULL, &A_trans);
494 fasp_dcsr_sort(&A_trans);
495 status = fasp_solver_umfpack(&A_trans, &b, &x, print_level);
496 fasp_dcsr_free(&A_trans);
497 fasp_dcsr_free(&Acsr);
498 if (status >= 0) status = 1; // Direct solver returns 1
499 }
500#endif
501
502#ifdef WITH_PARDISO // use PARDISO directly
503 else if (solver_type == SOLVER_PARDISO) {
504 dCSRmat Acsr = fasp_format_dbsr_dcsr(&A);
505 fasp_dcsr_sort(&Acsr);
506 status = fasp_solver_pardiso(&Acsr, &b, &x, print_level);
507 fasp_dcsr_free(&Acsr);
508 if (status >= 0) status = 1; // Direct solver returns 1
509 }
510#endif
511
512 else {
513 printf("### ERROR: Wrong solver type %d!!!\n", solver_type);
514 status = ERROR_SOLVER_TYPE;
515 }
516
517 if (print_level > PRINT_MIN) {
518 if (status < 0) {
519 cout << "\n### WARNING: Solver does not converge!\n" << endl;
520 } else {
521 cout << "\nSolver converges successfully!\n" << endl;
522 }
523 }
524
525 if (output_type) fclose(stdout);
526
527 return status;
528}
