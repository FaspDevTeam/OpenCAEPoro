137OCP_DBL FlowUnit_ODGW::CalKro_Stone2(const OCP_DBL& krow, const OCP_DBL& krog,
138 const OCP_DBL& krw, const OCP_DBL& krg) const
139{
140 // krog : oil relative permeability for a system with oil, gas and connate water
141 // krow : oil relative permeability for a system with oil and water only
142
143 OCP_DBL kro =
144 kroMax * ((krow / kroMax + krw) * (krog / kroMax + krg) - (krw + krg));
145 if (kro < 0) kro = 0;
146
147 return kro;
148}

151OCP_DBL FlowUnit_ODGW::CalKro_Default(const OCP_DBL& Sg, const OCP_DBL& Sw,
152 const OCP_DBL& krog, const OCP_DBL& krow) const
153{
154 OCP_DBL tmp = Sg + Sw - Swco;
155 if (tmp <= TINY) {
156 return kroMax;
157 }
158 OCP_DBL kro = (Sg * krog + (Sw - Swco) * krow) / tmp;
159 return kro;
160}

280OCP_DBL FlowUnit_ODGW01::CalKro_Stone2Der(OCP_DBL krow, OCP_DBL krog, OCP_DBL krw, OCP_DBL krg,
281 OCP_DBL dkrwdSw, OCP_DBL dkrowdSw, OCP_DBL dkrgdSg,
282 OCP_DBL dkrogdSg, OCP_DBL& out_dkrodSw,
283 OCP_DBL& out_dkrodSg) const
284{
285 OCP_DBL kro, dkrodSw, dkrodSg;
286 kro = kroMax * ((krow / kroMax + krw) * (krog / kroMax + krg) - (krw + krg));
287
288 dkrodSw =
289 kroMax * ((dkrowdSw / kroMax + dkrwdSw) * (krog / kroMax + krg) - (dkrwdSw));
290 dkrodSg =
291 kroMax * ((krow / kroMax + krw) * (dkrogdSg / kroMax + dkrgdSg) - (dkrgdSg));
292
293 if (kro < 0) {
294 kro = 0;
295 dkrodSg = 0;
296 dkrodSw = 0;
297 }
298 out_dkrodSw = dkrodSw;
299 out_dkrodSg = dkrodSg;
300 return kro;
301}

303OCP_DBL FlowUnit_ODGW01::CalKro_DefaultDer(const OCP_DBL& Sg, const OCP_DBL& Sw,
304 const OCP_DBL& krog, const OCP_DBL& krow,
305 const OCP_DBL& dkrogSg, const OCP_DBL& dkrowSw,
306 OCP_DBL& dkroSg, OCP_DBL& dkroSw) const
307{
308 OCP_DBL tmp = Sg + Sw - Swco;
309 if (tmp <= TINY) {
310 dkroSg = 0;
311 dkroSw = 0;
312 return kroMax;
313 }
314 OCP_DBL kro = (Sg * krog + (Sw - Swco) * krow) / tmp;
315 dkroSg = (krog + Sg * dkrogSg - kro) / tmp;
316 dkroSw = (krow + (Sw - Swco) * dkrowSw - kro) / tmp;
317 return kro;
318}

320void FlowUnit_ODGW01::Generate_SWPCWG()
321{
322 if (SGOF.IsEmpty()) OCP_ABORT("SGOF is missing!");
323 if (SWOF.IsEmpty()) OCP_ABORT("SWOF is missing!");
324
325 std::vector<OCP_DBL> Sw(SWOF.GetCol(0));
326 std::vector<OCP_DBL> Pcow(SWOF.GetCol(3));
327 USI n = Sw.size();
328 for (USI i = 0; i < n; i++) {
329 OCP_DBL Pcgo = SGOF.Eval(0, 1 - Sw[i], 3);
330 Pcow[i] += Pcgo; // Pcgw
331 }
332
333 SWPCGW.PushCol(Sw);
334 SWPCGW.PushCol(Pcow);
335 SWPCGW.SetRowCol();
336}

342void FlowUnit_ODGW01_Miscible::CalKrPc(const OCP_DBL* S_in, OCP_DBL* kr_out, OCP_DBL* pc_out,
343 const OCP_DBL& MySurTen, OCP_DBL& MyFk, OCP_DBL& MyFp)
344{
345 surTen = MySurTen;
346 if (surTen >= surTenRef || surTen < TINY) {
347 MyFk = 1; MyFp = 1;
348 FlowUnit_ODGW01::CalKrPc(S_in, kr_out, pc_out, surTen, surTen, surTen);
349 }
350 else {
351 OCP_DBL So = S_in[0];
352 OCP_DBL Sg = S_in[1];
353 OCP_DBL Sw = S_in[2];
354
355 Fk = min(1.0, pow(surTen / surTenRef, Fkexp));
356 Fp = min(surTenPc, surTen / surTenRef);
357
358 SWOF.Eval_All(0, Sw, data, cdata);
359 const OCP_DBL krw = data[1];
360 OCP_DBL krow = data[2];
361 const OCP_DBL Pcwo = -data[3];
362
363 SGOF.Eval_All(0, Sg, data, cdata);
364 OCP_DBL krg = data[1];
365 OCP_DBL krog = data[2];
366 const OCP_DBL Pcgo = data[3] * Fp;
367
368 OCP_DBL kro = CalKro_Stone2(krow, krog, krw, krg);
369 // OCP_DBL kro = CalKro_Default(Sg, Sw, krog, krow);
370 OCP_DBL krgt = SGOF.Eval(0, (1 - Sw), 1);
371 OCP_DBL krh = 0.5 * (krow + krgt);
372
373 // from CMG, see *SIGMA
374 kro = Fk * kro + (1 - Fk) * krh * So / (1 - Sw);
375 krg = Fk * krg + (1 - Fk) * krh * Sg / (1 - Sw);
376
377 kr_out[0] = kro;
378 kr_out[1] = krg;
379 kr_out[2] = krw;
380 pc_out[0] = 0;
381 pc_out[1] = Pcgo;
382 pc_out[2] = Pcwo;
383
384 MyFk = Fk;
385 MyFp = Fp;
386
387 // cout << Fk << endl;
388 }
389}

392void FlowUnit_ODGW01_Miscible::CalKrPcDeriv(const OCP_DBL* S_in, OCP_DBL* kr_out, OCP_DBL* pc_out,
393 OCP_DBL* dkrdS, OCP_DBL* dPcjdS, const OCP_DBL& MySurTen, OCP_DBL& MyFk, OCP_DBL& MyFp)
394{
395 surTen = MySurTen;
396 if (surTen >= surTenRef || surTen < TINY) {
397 MyFk = 1; MyFp = 1;
398 FlowUnit_ODGW01::CalKrPcDeriv(S_in, kr_out, pc_out, dkrdS, dPcjdS, surTen, surTen, surTen);
399 }
400 else {
401 const OCP_DBL So = S_in[0];
402 const OCP_DBL Sg = S_in[1];
403 const OCP_DBL Sw = S_in[2];
404
405 Fk = min(1.0, pow(surTen / surTenRef, Fkexp));
406 Fp = min(surTenPc, surTen / surTenRef);
407
408 SWOF.Eval_All(0, Sw, data, cdata);
409 const OCP_DBL krw = data[1];
410 const OCP_DBL dKrwdSw = cdata[1];
411 const OCP_DBL krow = data[2];
412 const OCP_DBL dKrowdSw = cdata[2];
413 const OCP_DBL Pcwo = -data[3];
414 const OCP_DBL dPcwdSw = -cdata[3];
415
416 SGOF.Eval_All(0, Sg, data, cdata);
417 OCP_DBL krg = data[1];
418 const OCP_DBL dKrgdSg = cdata[1];
419 const OCP_DBL krog = data[2];
420 const OCP_DBL dKrogdSg = cdata[2];
421 const OCP_DBL Pcgo = data[3] * Fp;
422 const OCP_DBL dPcgdSg = cdata[3] * Fp;
423
424 OCP_DBL dKrodSg{ 0 }, dKrodSw{ 0 }, kro{ 0 };
425 kro = CalKro_Stone2Der(krow, krog, krw, krg, dKrwdSw, dKrowdSw, dKrgdSg, dKrogdSg,
426 dKrodSw, dKrodSg);
427
428 OCP_DBL dkrgd1_Sw = 0;
429 const OCP_DBL krgt = SGOF.Eval(0, (1 - Sw), 1, dkrgd1_Sw);
430 const OCP_DBL krh = 0.5 * (krow + krgt);
431
432 // from CMG, see *SIGMA
433 kro = Fk * kro + (1 - Fk) * krh * So / (1 - Sw);
434 krg = Fk * krg + (1 - Fk) * krh * Sg / (1 - Sw);
435
436 kr_out[0] = kro;
437 kr_out[1] = krg;
438 kr_out[2] = krw;
439 pc_out[0] = 0;
440 pc_out[1] = Pcgo;
441 pc_out[2] = Pcwo;
442
443 OCP_DBL dkrhdSw = 0.5 * (dKrowdSw - dkrgd1_Sw);
444 OCP_DBL temp = (1 - Fk) / (1 - Sw) * (krh / (1 - Sw) + dkrhdSw);
445
446 dkrdS[0] = (1 - Fk) * krh / (1 - Sw);
447 dkrdS[1] = Fk * dKrodSg;
448 dkrdS[2] = Fk * dKrodSw + temp * So;
449 dkrdS[3] = 0;
450 dkrdS[4] = Fk * dKrgdSg + (1 - Fk) * krh / (1 - Sw);
451 dkrdS[5] = temp * Sg;
452 dkrdS[6] = 0;
453 dkrdS[7] = 0;
454 dkrdS[8] = dKrwdSw;
455
456 dPcjdS[0] = 0;
457 dPcjdS[1] = 0;
458 dPcjdS[2] = 0;
459 dPcjdS[3] = 0;
460 dPcjdS[4] = dPcgdSg;
461 dPcjdS[5] = 0;
462 dPcjdS[6] = 0;
463 dPcjdS[7] = 0;
464 dPcjdS[8] = dPcwdSw;
465
466 MyFk = Fk;
467 MyFp = Fp;
468
469 // cout << Fk << endl;

581OCP_DBL FlowUnit_ODGW02::CalKro_Stone2Der(OCP_DBL krow, OCP_DBL krog, OCP_DBL krw, OCP_DBL krg,
582 OCP_DBL dkrwdSw, OCP_DBL dkrowdSo, OCP_DBL dkrgdSg,
583 OCP_DBL dkrogdSo, OCP_DBL& out_dkrodSo) const
584{
585 OCP_DBL kro, dKrodSo;
586
587 kro = kroMax * ((krow / kroMax + krw) * (krog / kroMax + krg) - (krw + krg));
588 dKrodSo = dkrowdSo * (krog / kroMax + krg) + dkrogdSo * (krow / kroMax + krw);
589

593 }

598OCP_DBL FlowUnit_ODGW02::CalKro_DefaultDer(const OCP_DBL& Sg, const OCP_DBL& Sw,
599 const OCP_DBL& krog, const OCP_DBL& krow,
600 const OCP_DBL& dkrogdSo, const OCP_DBL& dkrowdSo,
601 OCP_DBL& out_dkrodSo) const
602{
603 OCP_DBL tmp = Sg + Sw - Swco;
604 OCP_DBL kro = (Sg * krog + (Sw - Swco) * krow) / tmp;

606
607 if (tmp <= TINY) {
608 kro = kroMax;
609 out_dkrodSo = 0;
610 }
611 return kro;
