134OCP_DBL FlowUnit_ODGW::CalKro_Stone2(const OCP_DBL& krow, const OCP_DBL& krog,
135 const OCP_DBL& krw, const OCP_DBL& krg) const
136{
137 // krog : oil relative permeability for a system with oil, gas and connate water
138 // krow : oil relative permeability for a system with oil and water only
139
140 OCP_DBL kro =
141 kroMax * ((krow / kroMax + krw) * (krog / kroMax + krg) - (krw + krg));
142 if (kro < 0) kro = 0;
143
144 return kro;
145}

148OCP_DBL FlowUnit_ODGW::CalKro_Default(const OCP_DBL& Sg, const OCP_DBL& Sw,
149 const OCP_DBL& krog, const OCP_DBL& krow) const
150{
151 OCP_DBL tmp = Sg + Sw - Swco;
152 if (tmp <= TINY) {
153 return kroMax;
154 }
155 OCP_DBL kro = (Sg * krog + (Sw - Swco) * krow) / tmp;
156 return kro;

267OCP_DBL FlowUnit_ODGW01::CalKro_Stone2Der(OCP_DBL krow, OCP_DBL krog, OCP_DBL krw, OCP_DBL krg,
268 OCP_DBL dkrwdSw, OCP_DBL dkrowdSw, OCP_DBL dkrgdSg,
269 OCP_DBL dkrogdSg, OCP_DBL& out_dkrodSw,
270 OCP_DBL& out_dkrodSg) const
271{
272 OCP_DBL kro, dkrodSw, dkrodSg;
273 kro = kroMax * ((krow / kroMax + krw) * (krog / kroMax + krg) - (krw + krg));
274
275 dkrodSw =
276 kroMax * ((dkrowdSw / kroMax + dkrwdSw) * (krog / kroMax + krg) - (dkrwdSw));
277 dkrodSg =
278 kroMax * ((krow / kroMax + krw) * (dkrogdSg / kroMax + dkrgdSg) - (dkrgdSg));
279
280 if (kro < 0) {
281 kro = 0;
282 dkrodSg = 0;
283 dkrodSw = 0;
284 }
285 out_dkrodSw = dkrodSw;
286 out_dkrodSg = dkrodSg;
287 return kro;
288}

290OCP_DBL FlowUnit_ODGW01::CalKro_DefaultDer(const OCP_DBL& Sg, const OCP_DBL& Sw,
291 const OCP_DBL& krog, const OCP_DBL& krow,
292 const OCP_DBL& dkrogSg, const OCP_DBL& dkrowSw,
293 OCP_DBL& dkroSg, OCP_DBL& dkroSw) const
294{
295 OCP_DBL tmp = Sg + Sw - Swco;
296 if (tmp <= TINY) {
297 dkroSg = 0;
298 dkroSw = 0;
299 return kroMax;
300 }
301 OCP_DBL kro = (Sg * krog + (Sw - Swco) * krow) / tmp;
302 dkroSg = (krog + Sg * dkrogSg - kro) / tmp;
303 dkroSw = (krow + (Sw - Swco) * dkrowSw - kro) / tmp;
304 return kro;
305}
