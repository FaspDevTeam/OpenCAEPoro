25void Solver::RunSimulation(Reservoir& rs, OCPControl& ctrl, OCPOutput& output)
26{
27 GetWallTime timer;
28 timer.Start();
29
30 USI numTSteps = ctrl.GetNumTSteps();
31 output.PrintInfoSched(rs, ctrl, timer.Stop());
32
33 for (USI d = 0; d < numTSteps - 1; d++) {
34 rs.ApplyControl(d);
35 ctrl.ApplyControl(d);
36 ctrl.InitTime(d);
37 while (!ctrl.IsCriticalTime(d + 1)) {
38 GoOneStep(rs, ctrl);
39 output.SetVal(rs, ctrl);
40 }
41 output.PrintInfoSched(rs, ctrl, timer.Stop());
42 }
43
44 ctrl.RecordTotalTime(timer.Stop() / 1000);
45}

48void Solver::GoOneStep(Reservoir& rs, OCPControl& ctrl)
49{
50 OCP_DBL& dt = ctrl.GetCurDt();
51
52 //#ifdef DEBUG
53 // cout << "### DEBUG: " << fixed << ctrl.GetCurTime() << " Days";
54 // cout << " NR: " << ctrl.GetNRiterT() << " LS: " << ctrl.GetLSiterT() << endl;
55 //#endif // DEBUG
56
57 // Prepare for time marching
58 Prepare(rs, dt);
59
60 // Time marching with adaptive time stepsize
61 while (true) {
62 if (dt < MIN_TIME_CURSTEP) OCP_ABORT("Time stepsize is too small!");
63 AssembleSolve(rs, ctrl);
64 if (!UpdateProperty(rs, ctrl)) {
65 ctrl.ResetIterNRLS();
66 continue;
67 }
68 if (FinishNR(rs, ctrl)) break;
69 }
70
71 // Finish current time step
72 FinishStep(rs, ctrl);
73}
