14BOMixture::BOMixture(const ParamReservoir& rs_param, const USI& PVTmode, const USI& i)
15{
16 mixtureType = BLKOIL;
17 mode = PVTmode;
18 numPhase = rs_param.numPhase;
19 numCom = rs_param.numCom;
20
21 Allocate();
22
23 if (rs_param.density.activity) {
24 std_RhoO = rs_param.density.data[0];
25 std_RhoW = rs_param.density.data[1];
26 std_RhoG = rs_param.density.data[2];
27 } else {
28 std_RhoO = (141.5 * RHOW_STD) / (rs_param.gravity.data[0] + 131.5);
29 std_RhoW = RHOW_STD * rs_param.gravity.data[1];
30 std_RhoG = RHOAIR_STD * rs_param.gravity.data[2];
31 }
32
33 std_GammaO = GRAVITY_FACTOR * std_RhoO;
34 std_GammaG = GRAVITY_FACTOR * std_RhoG;
35 std_GammaW = GRAVITY_FACTOR * std_RhoW;
36
37 if (rs_param.PVTW_T.data.size() != 0) {
38 PVTW.Setup(rs_param.PVTW_T.data[i]);
39 len = max(len, PVTW.GetColNum());
40 }
41 if (rs_param.PVCO_T.data.size() != 0) {
42 PVCO.Setup(rs_param.PVCO_T.data[i]);
43 len = max(len, PVCO.GetColNum());
44 }

46 PVDO.Setup(rs_param.PVDO_T.data[i]);

55}

190OCP_DBL BOMixture::GammaPhaseW(const OCP_DBL& Pin)
191{
192
193 PVTW.Eval_All(0, Pin, data, cdata);
194 OCP_DBL Pw0 = data[0];
195 OCP_DBL bw0 = data[1];
196 OCP_DBL cbw = data[2];
197 OCP_DBL bw = (bw0 * (1 - cbw * (Pin - Pw0)));
198
199 return std_GammaW / bw;
200}

203{
