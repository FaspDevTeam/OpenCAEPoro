24// -- (1) Read input data;
25// -- (2) Setup static information;
26// -- (3) Initialize reservoir status;
27// -- (4) Run dynamic simulation;
28// -- (5) Output summary and other results.
29int main(int argc, const char* argv[])
30{
31 if (argc <= 1) {
32 // Provide at least InputFileName for the input data
33 cout << "Wrong number of arguments. Usage: " << argv[0] << " <InputFileName>"
34 << endl;
35 return -1;
36 }
37
38 // Step 0. Print simulator version information.
39 OpenCAEPoro simulator;
40 simulator.PrintVersion();
41
42 // Step 1. Read params from an input file to internal params data structure.
43 // Note: The keywords are almost compatible with Ecl simulator; see Keywords.md.
44 ParamRead rp;
45 rp.ReadInputFile(argv[1]);
46
47 // Step 2. Set params using command-line and internal params data structure, and
48 // setup static information, such as active grids, and connections between them.
49 // Note: Memory allocation for linear systems will also be done at this time.
50 simulator.SetupSimulator(rp, argc, argv);
51
52 // Step 3. Initialize the reservoir, which finishs the first step in iterations.
53 // For example: initial pressure, saturations, moles of components, and initial
54 // guess of well pressure.
55 simulator.InitReservoir();
56
57 // Step 4. Run dynamic simulation using methods like IMPEC and FIM. It's a
58 // combination of functions of various modules which you could make changes
59 // whenever necessary.
60 simulator.RunSimulation();
61
62 // Step 5. Output the results according to params users give. It will generate
63 // a summary file in your input data directory.
64 simulator.OutputResults();
65
66 return 0;
67}
