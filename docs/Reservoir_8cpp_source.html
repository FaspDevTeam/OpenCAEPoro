466void Reservoir::PrintSolFIM(const string& outfile) const
467{
468 ofstream outu(outfile);
469 if (!outu.is_open()) cout << "Can not open " << outfile << endl;
470 const OCP_USI nb = bulk.numBulk;
471 const OCP_USI nc = bulk.numCom;
472
473 for (OCP_USI n = 0; n < nb; n++) {
474 // Pressure
475 outu << bulk.P[n] << "\n";
476 // Ni
477 for (USI i = 0; i < nc; i++) {
478 outu << bulk.Ni[n * nc + i] << "\n";
479 }
480 }
481 // Well Pressure
482 for (USI w = 0; w < allWells.numWell; w++) {
483 outu << allWells.GetWBHP(w) << "\n";
484 }
485 outu.close();
486}
