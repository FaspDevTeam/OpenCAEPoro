55bool OCP_IMPEC::UpdateProperty(Reservoir& rs, OCPControl& ctrl)
56{
57 OCP_DBL& dt = ctrl.current_dt;
58
59 // first check : Pressure check
60 OCP_INT flagCheck = rs.CheckP();
61 switch (flagCheck) {
62 case 1:
63 cout << "well change" << endl;
64 dt /= 2;
65 rs.ResetVal00IMPEC();
66 return false;
67 case 2:
68 cout << "well change" << endl;
69 dt /= 1;
70 rs.ResetVal00IMPEC();
71 // rs.ResetWellIMPEC();
72 return false;
73 default:
74 break;
75 }
76
77 rs.CalFLuxIMPEC();
78
79 // second check : CFL check
80 OCP_DBL cfl = rs.CalCFL01IMPEC(dt);
81 if (cfl > 1) {
82 dt /= 2;
83 rs.ResetVal01IMPEC();
84 cout << "CFL is too big" << endl;
85 return false;
86 }
87
88 rs.MassConseveIMPEC(dt);
89
90 // third check: Ni check
91 if (!rs.CheckNi()) {
92 dt /= 2;
93 rs.ResetVal02IMPEC();
94 cout << "Negative Ni occurs\n";
95 return false;
96 }
97
98 rs.CalFlashIMPEC();
99 rs.CalVpore();
100
101 // fouth check: Volume error check
102 if (!rs.CheckVe(0.01)) {
103 dt /= 2;
104 rs.ResetVal03IMPEC();
105 return false;
106 }
107
108 rs.CalKrPc();
109 rs.CalConnFluxIMPEC();
110 return true;
111}

201bool OCP_FIM::FinishNR(Reservoir& rs, OCPControl& ctrl)
202{
203 if (ctrl.iterNR > ctrl.ctrlNR.maxNRiter) {
204 ctrl.current_dt *= ctrl.ctrlTime.cutFacNR;
205 rs.ResetFIM(false);
206 rs.CalResFIM(resFIM, ctrl.current_dt);
207 resFIM.maxRelRes0_v = resFIM.maxRelRes_v;
208 ctrl.ResetIterNRLS();
209 cout << "NR Failed, Cut time Step and reset!\n";
210 return false;
211 }
212
213 OCP_DBL NRdPmax = rs.GetNRdPmax();
214 OCP_DBL NRdSmax = rs.GetNRdSmax();
215
216#ifdef DEBUG
217 cout << "### DEBUG: Residuals = " << scientific << resFIM.maxRelRes0_v << " "
218 << resFIM.maxRelRes_v << " " << resFIM.maxRelRes_mol << " " << NRdSmax
219 << " " << NRdPmax << endl;
220#endif
221
222 if (resFIM.maxRelRes_v <= resFIM.maxRelRes0_v * ctrl.ctrlNR.NRtol ||
223 resFIM.maxRelRes_v <= ctrl.ctrlNR.NRtol ||
224 resFIM.maxRelRes_mol <= ctrl.ctrlNR.NRtol ||
225 (NRdPmax <= ctrl.ctrlNR.NRdPmin && NRdSmax <= ctrl.ctrlNR.NRdSmin)) {
226
227 OCP_INT flagCheck = rs.CheckP(false, true);
228#if DEBUG
229 if (flagCheck > 0) {
230 cout << ">> Switch well constraint: Case " << flagCheck << endl;
231 }
232#endif
233
234 switch (flagCheck) {
235 case 1:
236 ctrl.current_dt *= ctrl.ctrlTime.cutFacNR;
237 rs.ResetFIM(true);
238 rs.CalResFIM(resFIM, ctrl.current_dt);
239 resFIM.maxRelRes0_v = resFIM.maxRelRes_v;
240 ctrl.ResetIterNRLS();
241 return false;
242 case 2:
243 ctrl.current_dt /= 1;
244 rs.ResetFIM(true);
245 rs.CalResFIM(resFIM, ctrl.current_dt);
246 resFIM.maxRelRes0_v = resFIM.maxRelRes_v;
247 ctrl.ResetIterNRLS();
248 return false;
249 default:
250 break;
251 }
252 return true;
253 } else {
254 return false;
255 }
