181void ParamReservoir::setVal(vector<T>& obj, const T& val, const vector<USI>& index)
182{
183 USI Nx = dimens.nx;
184 USI Ny = dimens.ny;
185 OCP_USI NxNy = Nx * Ny;
186 OCP_USI id = 0;
187
188 for (USI k = index[4]; k <= index[5]; k++) {
189 for (USI j = index[2]; j <= index[3]; j++) {
190 for (USI i = index[0]; i <= index[1]; i++) {
191 id = k * NxNy + j * Nx + i;
192 obj[id] = val;
193 }
194 }
195 }

277void ParamReservoir::InputEQUALS(ifstream& ifs)
278{
279 vector<USI> index(6, 0);
280 vector<string> vbuf;
281
282 while (ReadLine(ifs, vbuf)) {
283 if (vbuf[0] == "/") break;
284
285 index[0] = 0, index[1] = dimens.nx - 1;
286 index[2] = 0, index[3] = dimens.ny - 1;
287 index[4] = 0, index[5] = dimens.nz - 1;
288
289 string objName = vbuf[0];
290 OCP_DBL val = stod(vbuf[1]);
291
292 DealDefault(vbuf);
293
294 for (USI n = 2; n < 8; n++) {
295 if (vbuf[n] != "DEFAULT") index[n - 2] = stoi(vbuf[n]) - 1;
296 }
297 if (index[0] < 0 || index[2] < 0 || index[4] < 0 || index[1] > dimens.nx - 1 ||
298 index[3] > dimens.ny - 1 || index[5] > dimens.nz - 1) {
299 string MyError = "WRONG Range in " + objName + " in EQUALS!";
300 OCP_ABORT("WRONG Range in ");
301 }
302
303 vector<OCP_DBL>* objPtr = FindPtr(objName);
304
305 if (objPtr != nullptr) {
306 if (objName == "TOPS") {
307 objPtr->resize(dimens.nx * dimens.ny);
308 index[4] = index[5] = 0;
309 } else {
310 objPtr->resize(numGrid);
311 }
312 setVal(*objPtr, val, index);
313 } else {
314 OCP_ABORT("Wrong object name: " + objName);
315 }
316 }
317
318 std::cout << SATNUM.activity << endl;
319 std::cout << PVTNUM.activity << endl;
320 cout << "EQUALS" << endl;
321}

586void ParamReservoir::CheckGrid()
587{
588 if (coord.size() == 0) {
589 if (tops.size() != dimens.nx * dimens.ny) OCP_ABORT("Wrong TOPS size!");
590 if (dx.size() != numGrid) OCP_ABORT("Wrong DX size!");
591 if (dy.size() != numGrid) OCP_ABORT("Wrong DY size!");
592 if (dz.size() != numGrid) OCP_ABORT("Wrong DZ size!");
593 } else {
594 if (coord.size() != (dimens.nx + 1) * (dimens.ny + 1) * 6)
595 OCP_ABORT("Wrong COORD size!");
596 if (zcorn.size() != numGrid * 8) OCP_ABORT("Wrong ZCORN size!");
597 }
598
599 if (poro.size() != numGrid) OCP_ABORT("Wrong PORO size!");
600 if (permX.size() != numGrid) OCP_ABORT("Wrong PERMX size!");
601 if (permY.size() != numGrid) OCP_ABORT("Wrong PERMY size!");
602 if (permZ.size() != numGrid) OCP_ABORT("Wrong PERMZ size!");
603 if (ntg.size() != numGrid) {
604 ntg.resize(numGrid, 1);
605 cout << "Reset Ntg size to 1!" << endl;
606 }
607}
