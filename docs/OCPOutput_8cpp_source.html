519 for (USI i = 1; i < col; i++) {
520 outF << "\t" << setw(ns) << Sumdata[id++].Obj;
521 if (id == num) break;

523 outF << endl;
524

527

540
541 outF << endl;

634void DetailInfo::PrintInfo(const string& dir, const Reservoir& rs,
635 const OCP_DBL& days) const
636{
637 string FileOut = dir + "RPT.out";
638 ofstream outF;
639 outF.open(FileOut, ios::app);
640 if (!outF.is_open()) {
641 OCP_ABORT("Can not open " + FileOut);
642 }
643
644 USI nx = rs.grid.GetGridNx();
645 USI ny = rs.grid.GetGridNy();
646 OCP_USI num = rs.grid.GetGridNum();
647 OCP_USI bId;
648
649 // PRESSURE
650 if (PRE) {
651 outF << "PRESSURE : psia"
652 << " ";
653 outF << fixed << setprecision(3) << days << " DAYS";
654 for (OCP_USI i = 0; i < num; i++) {
655 if (i % nx == 0) outF << "\n";
656 if (i % (nx * ny) == 0) outF << "\n";
657 if (rs.grid.MapG2B(i).IsAct()) {
658 bId = rs.grid.MapG2B(i).GetId();
659 outF << fixed << setprecision(3) << rs.bulk.P[bId] << " ";
660 } else {
661 outF << "N ";
662 }
663 }
664 outF << "\n\n";
665 }
666
667 outF.close();
