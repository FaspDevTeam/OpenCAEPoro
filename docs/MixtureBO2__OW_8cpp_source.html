24 // Water Properties
25 PVTW.Eval_All(0, P, data, cdata);
26 OCP_DBL Pw0 = data[0];
27 OCP_DBL bw0 = data[1];
28 OCP_DBL cbw = data[2];

42
43 mu[0] = data[2];
44 xi[0] = 1 / (CONV1 * bo);
45 rho[0] = std_RhoO / bo;
46 Ni[0] = Vpore * (1 - S[1]) * xi[0];
47
48 xij[0 * 2 + 0] = 1; xij[0 * 2 + 1] = 0;
49 xij[1 * 2 + 0] = 0; xij[1 * 2 + 1] = 1;
50
51 v[0] = CONV1 * Ni[0] * bo;
52 v[1] = CONV1 * Ni[1] * bw;
53 vf = v[0] + v[1];
54 S[0] = v[0] / vf;
55 S[1] = v[1] / vf;
56 vfi[0] = CONV1 * bo;
57 vfi[1] = CONV1 * bw;
58 vfp = CONV1 * Ni[0] * bop + CONV1 * Ni[1] * bwp;

105void BOMixture::BOFlash_Ni_OW_Deriv(const OCP_DBL& Pin, const OCP_DBL* Niin)
106{
107 phaseExist.assign(numPhase, true);
108 dSec_dPri.assign(dSec_dPri.size(), 0);
109
110 P = Pin;
111 Ni[0] = Niin[0];
112 Ni[1] = Niin[1];
113 Nt = Ni[0] + Ni[1];
114
115 // Water Properties
116 PVTW.Eval_All(0, P, data, cdata);
117 OCP_DBL Pw0 = data[0];
118 OCP_DBL bw0 = data[1];
119 OCP_DBL cbw = data[2];
120 OCP_DBL bw = bw0 * (1 - cbw * (P - Pw0));
121 OCP_DBL bwp = -cbw * bw0;
122
123 mu[1] = data[3];
124 xi[1] = 1 / (bw * CONV1);
125 rho[1] = std_RhoW / bw;
126
127 muP[1] = cdata[3];
128 xiP[1] = -bwp / (bw * bw * CONV1);
129 rhoP[1] = CONV1 * xiP[1] * std_RhoW;
130
131 // Oil Properties
132 PVDO.Eval_All(0, P, data, cdata);
133 OCP_DBL bo = data[1];
134 OCP_DBL bop = cdata[1];
135
136 mu[0] = data[2];
137 xi[0] = 1 / (CONV1 * bo);
138 rho[0] = std_RhoO / bo;
139
140 muP[0] = cdata[2];
141 xiP[0] = -xi[0] * bop / bo;
142 rhoP[0] = -rho[0] * bop / bo;
143
144 xij[0 * 2 + 0] = 1; xij[0 * 2 + 1] = 0;
145 xij[1 * 2 + 0] = 0; xij[1 * 2 + 1] = 1;
146
147 v[0] = CONV1 * Ni[0] * bo;
148 v[1] = CONV1 * Ni[1] * bw;
149 vf = v[0] + v[1];
150 S[0] = v[0] / vf;
151 S[1] = v[1] / vf;
152 vfi[0] = CONV1 * bo;
153 vfi[1] = CONV1 * bw;
154 vfp = CONV1 * Ni[0] * bop + CONV1 * Ni[1] * bwp;
155
156 dSec_dPri[0] = (CONV1 * Ni[0] * bop - S[0] * vfp) / vf; // dSo / dP
157 dSec_dPri[1] = (CONV1 * bo - S[0] * vfi[0]) / vf; // dSo / dNo
158 dSec_dPri[2] = -S[0] * vfi[1] / vf; // dSo / dNw
159
160 dSec_dPri[3] = (CONV1 * Ni[1] * bwp - S[1] * vfp) / vf; // dSw / dP
161 dSec_dPri[4] = -S[1] * vfi[0] / vf; // dSw / dNo
162 dSec_dPri[5] = (CONV1 * bw - S[1] * vfi[1]) / vf; // dSw / dNw
163
164}

167OCP_DBL BOMixture::XiPhase_OW(const OCP_DBL& Pin, const OCP_DBL* Ziin)
168{
169 if (Ziin[1] > 1 - TINY) {
170 // inj fluid is water
171 PVTW.Eval_All(0, Pin, data, cdata);
172 OCP_DBL Pw0 = data[0];
173 OCP_DBL bw0 = data[1];
174 OCP_DBL cbw = data[2];
175 OCP_DBL bw = bw0 * (1 - cbw * (P - Pw0));
176 OCP_DBL xiw = 1 / (CONV1 * bw);
177 return xiw;
178 }
179 else {
180 OCP_ABORT("Wrong Zi!");
181 }
182}
