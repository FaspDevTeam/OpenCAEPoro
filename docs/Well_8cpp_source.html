676 for (USI k = 0; k < seg_num; k++) {
677 myBulk.flashCal[pvtnum]->Flash_Ni(Ptmp, myBulk.T, tmpNi.data());
678 for (USI j = 0; j < myBulk.numPhase; j++) {
679 if (myBulk.flashCal[pvtnum]->phaseExist[j]) {
680 rhotmp = myBulk.flashCal[pvtnum]->rho[j];
681 qtacc += myBulk.flashCal[pvtnum]->v[j] / seg_num;
682 rhoacc += myBulk.flashCal[pvtnum]->v[j] * rhotmp *
683 GRAVITY_FACTOR / seg_num;
684#ifdef _DEBUG
685 if (rhotmp <= 0 || !isfinite(rhotmp)) {
686 OCP_ABORT("Wrong rho " + to_string(rhotmp));
687 }
688#endif // _DEBUG
689 }
690 }
691 Ptmp -= rhoacc / qtacc * seg_len;
692 }
693 dGperf[p] = Pperf - Ptmp;
694 }
695 dG[0] = dGperf[0];
696 for (USI p = 1; p < numPerf; p++) {
697 dG[p] = dG[p - 1] + dGperf[p];
698 }
699 } else if (depth >= perf.back().depth) {
700 // Well is lower
701
702 // check qi_lbmol ---- test
703 if (perf[0].state == CLOSE) {
704 for (USI p = 1; p <= numPerf; p++) {
705 if (perf[p].state == OPEN) {
706 for (USI i = 0; i < nc; i++) {
707 perf[numPerf - 1].qi_lbmol[i] = perf[p].qi_lbmol[i];
708 }
709 break;
710 }
711 }
712 }
713
714 for (USI p = 0; p < numPerf; p++) {
715 if (p == numPerf - 1) {
716 seg_num = ceil((depth - perf[numPerf - 1].depth) / maxlen);
717 seg_len = (depth - perf[numPerf - 1].depth) / seg_num;
718 } else {
719 seg_num = ceil((perf[p + 1].depth - perf[p].depth) / maxlen);
720 seg_len = (perf[p + 1].depth - perf[p].depth) / seg_num;
721 }
722
723 OCP_USI n = perf[p].location;
724 perf[p].P = BHP + dG[p];
725 OCP_DBL Pperf = perf[p].P;
726 OCP_DBL Ptmp = Pperf;
727
728 USI pvtnum = myBulk.PVTNUM[n];
729 tmpNi.assign(nc, 0);
730 for (OCP_INT p1 = numPerf - 1; p1 - p >= 0; p1--) {
731 for (USI i = 0; i < nc; i++) {
732 tmpNi[i] += perf[p1].qi_lbmol[i];
733 }
734 }
735
736 // check tmpNi
737 for (auto& v : tmpNi) {
738 v = fabs(v);
739 }
740
741 for (USI k = 0; k < seg_num; k++) {
742 myBulk.flashCal[pvtnum]->Flash_Ni(Ptmp, myBulk.T, tmpNi.data());
743 for (USI j = 0; j < np; j++) {
744 if (myBulk.flashCal[pvtnum]->phaseExist[j]) {
745 rhotmp = myBulk.flashCal[pvtnum]->rho[j];
746 qtacc += myBulk.flashCal[pvtnum]->v[j] / seg_num;
747 rhoacc += myBulk.flashCal[pvtnum]->v[j] * rhotmp *
748 GRAVITY_FACTOR / seg_num;
749 }
750 }
751 Ptmp += rhoacc / qtacc * seg_len;
752 }
753 dGperf[p] = Ptmp - Pperf;
754 }
755 dG[numPerf - 1] = dGperf[numPerf - 1];
756 for (OCP_INT p = numPerf - 2; p >= 0; p--) {
757 dG[p] = dG[p + 1] + dGperf[p];
758 }
759
760 } else {
761 OCP_ABORT("Wrong well position!");
762 }

1191void Well::AssembleMatINJ_FIM(const Bulk& myBulk, LinearSystem& myLS,
1192 const OCP_DBL& dt) const
1193{
1194 OCP_FUNCNAME;
1195
1196 const OCP_USI wId = myLS.dim;
1197 // important !
1198 myLS.dim++;
1199
1200 const USI np = myBulk.numPhase;
1201 const USI nc = myBulk.numCom;
1202 const USI ncol = nc + 1;
1203 const USI ncol2 = np * nc + np;
1204 const USI bsize = ncol * ncol;
1205 const USI bsize2 = ncol * ncol2;
1206
1207 OCP_DBL mu, muP;
1208 OCP_DBL dP;
1209 OCP_DBL transIJ;
1210
1211 OCP_USI n_np_j;
1212
1213 vector<OCP_DBL> bmat(bsize, 0);
1214 vector<OCP_DBL> bmat2(bsize, 0);
1215 vector<OCP_DBL> dQdXpB(bsize, 0);
1216 vector<OCP_DBL> dQdXpW(bsize, 0);
1217 vector<OCP_DBL> dQdXsB(bsize2, 0);
1218
1219 for (USI p = 0; p < numPerf; p++) {
1220 OCP_USI n = perf[p].location;
1221 dQdXpB.assign(bsize, 0);
1222 dQdXpW.assign(bsize, 0);
1223 dQdXsB.assign(bsize2, 0);
1224
1225 dP = myBulk.P[n] - BHP - dG[p];
1226
1227 for (USI j = 0; j < np; j++) {
1228 n_np_j = n * np + j;
1229 if (!myBulk.phaseExist[n_np_j]) continue;
1230
1231 mu = myBulk.mu[n_np_j];
1232 muP = myBulk.muP[n_np_j];
1233
1234 for (USI i = 0; i < nc; i++) {
1235 // dQ / dP
1236 transIJ = perf[p].transj[j] * perf[p].xi * opt.zi[i];
1237 dQdXpB[(i + 1) * ncol] += transIJ * (1 - dP * muP / mu);
1238 dQdXpW[(i + 1) * ncol] += -transIJ;
1239
1240 // dQ / dS
1241 for (USI k = 0; k < np; k++) {
1242 dQdXsB[(i + 1) * ncol2 + k] +=
1243 CONV1 * CONV2 * perf[p].WI * perf[p].multiplier * perf[p].xi *
1244 opt.zi[i] * myBulk.dKr_dS[n * np * np + j * np + k] * dP / mu;
1245 }
1246 // dQ / dCij
1247 for (USI k = 0; k < nc; k++) {
1248 dQdXsB[(i + 1) * ncol2 + np + j * nc + k] +=
1249 -transIJ * dP / mu * myBulk.mux[n_np_j * nc + k];
1250 }
1251 }
1252 }
1253 // Bulk to Well
1254 bmat = dQdXpB;
1255 DaABpbC(ncol, ncol, ncol2, 1, dQdXsB.data(), &myBulk.dSec_dPri[n * bsize2], 1,
1256 bmat.data());
1257 Dscalar(bsize, dt, bmat.data());
1258 // Add
1259 USI ptr = myLS.diagPtr[n];
1260 for (USI i = 0; i < bsize; i++) {
1261 myLS.val[n][ptr * bsize + i] += bmat[i];
1262 }
1263 // Insert
1264 bmat = dQdXpW;
1265 Dscalar(bsize, dt, bmat.data());
1266 myLS.val[n].insert(myLS.val[n].end(), bmat.begin(), bmat.end());
1267 myLS.colId[n].push_back(wId);
1268
1269 // Well
1270 switch (opt.optMode) {
1271 case RATE_MODE:
1272 case ORATE_MODE:
1273 case GRATE_MODE:
1274 case WRATE_MODE:
1275 case LRATE_MODE:
1276 // Diag
1277 bmat.assign(bsize, 0);
1278 for (USI i = 0; i < nc; i++) {
1279 bmat[0] += dQdXpW[(i + 1) * ncol];
1280 bmat[(i + 1) * ncol + i + 1] = 1;
1281 }
1282 for (USI i = 0; i < bsize; i++) {
1283 myLS.diagVal[wId * bsize + i] += bmat[i];
1284 }
1285
1286 // OffDiag
1287 bmat = dQdXpB;
1288 DaABpbC(ncol, ncol, ncol2, 1, dQdXsB.data(),
1289 &myBulk.dSec_dPri[n * bsize2], 1, bmat.data());
1290 bmat2.assign(bsize, 0);
1291 for (USI i = 0; i < nc; i++) {
1292 Daxpy(ncol, opt.zi[i], bmat.data() + (i + 1) * ncol, bmat2.data());
1293 }
1294 myLS.val[wId].insert(myLS.val[wId].end(), bmat2.begin(), bmat2.end());
1295 myLS.colId[wId].push_back(n);
1296 break;
1297
1298 case BHP_MODE:
1299 // Diag
1300 bmat.assign(bsize, 0);
1301 for (USI i = 0; i < nc + 1; i++) {
1302 bmat[i * ncol + i] = 1;
1303 }
1304 // Add
1305 for (USI i = 0; i < bsize; i++) {
1306 myLS.diagVal[wId * bsize + i] += bmat[i];
1307 }
1308 // OffDiag
1309 bmat.assign(bsize, 0);
1310 // Insert
1311 myLS.val[wId].insert(myLS.val[wId].end(), bmat.begin(), bmat.end());
1312 myLS.colId[wId].push_back(n);
1313 // Solution
1314 // myLS.u[wId * ncol] = opt.maxBHP - BHP;
1315 break;
1316
1317 default:
1318 OCP_ABORT("Wrong Well Opt mode!");
1319 break;
1320 }
1321 }
1322 assert(myLS.val[wId].size() == numPerf * bsize);
1323 // Well self

1361 dQdXpB.assign(bsize, 0);

1379 dQdXpB[(i + 1) * ncol] +=
1380 transIJ * (1 - dP * muP / mu) + dP * perf[p].transj[j] * xij * xiP;
1381 dQdXpW[(i + 1) * ncol] += -transIJ;

1390 xi * xij * myBulk.dKr_dS[n * np * np + j * np + k];

1392 tmp += transIJ * myBulk.dPcj_dS[n * np * np + j * np + k];
1393 dQdXsB[(i + 1) * ncol2 + k] += tmp;
1394 }
1395 // dQ / dCij
1396 for (USI k = 0; k < nc; k++) {
1397 tmp = dP * perf[p].transj[j] * xij *
1398 (myBulk.xix[n_np_j * nc + k] -
1399 xi / mu * myBulk.mux[n_np_j * nc + k]);
1400 if (k == i) {
1401 tmp += perf[p].transj[j] * xi * dP;

1404 }
1405 }
1406 }

1408 bmat = dQdXpB;
1409 DaABpbC(ncol, ncol, ncol2, 1, dQdXsB.data(), &myBulk.dSec_dPri[n * bsize2], 1,

1424 Dscalar(bsize, dt, bmat.data());

1430 case RATE_MODE:
1431 case ORATE_MODE:
1432 case GRATE_MODE:
1433 case WRATE_MODE:
1434 case LRATE_MODE:
1435 // Diag
1436 bmat.assign(bsize, 0);
1437 for (USI i = 0; i < nc; i++) {
1438 bmat[0] += dQdXpW[(i + 1) * ncol] * opt.zi[i];

1440 }
1441 for (USI i = 0; i < bsize; i++) {
1442 myLS.diagVal[wId * bsize + i] += bmat[i];
1443 }
1444

1446 bmat = dQdXpB;
1447 DaABpbC(ncol, ncol, ncol2, 1, dQdXsB.data(),
1448 &myBulk.dSec_dPri[n * bsize2], 1, bmat.data());

1450 for (USI i = 0; i < nc; i++) {
1451 Daxpy(ncol, opt.zi[i], bmat.data() + (i + 1) * ncol, bmat2.data());

1453 myLS.val[wId].insert(myLS.val[wId].end(), bmat2.begin(), bmat2.end());
1454 myLS.colId[wId].push_back(n);

1456
1457 case BHP_MODE:
