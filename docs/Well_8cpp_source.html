223void Well::CalWI_Peaceman_Vertical(const Bulk& myBulk)
224{
225 OCP_FUNCNAME;
226
227 // this fomular needs to be carefully checked !
228 // especially the dz
229
230 for (USI p = 0; p < numPerf; p++) {
231 if (perf[p].WI > 0) {
232 break;
233 } else {
234 OCP_USI Idb = perf[p].location;
235 OCP_DBL dx = myBulk.dx[Idb];
236 OCP_DBL dy = myBulk.dy[Idb];
237 OCP_DBL dz = myBulk.dz[Idb] * myBulk.ntg[Idb];
238 OCP_DBL ro = 0;
239 switch (perf[p].direction) {
240 case X_DIRECTION: {
241 OCP_DBL kykz = myBulk.rockKy[Idb] * myBulk.rockKz[Idb];
242 OCP_DBL ky_kz = myBulk.rockKy[Idb] / myBulk.rockKz[Idb];
243 assert(kykz > 0);
244 ro =
245 0.28 *
246 pow((dy * dy * pow(1 / ky_kz, 0.5) + dz * dz * pow(ky_kz, 0.5)),
247 0.5);
248 ro /= (pow(ky_kz, 0.25) + pow(1 / ky_kz, 0.25));
249
250 if (perf[p].kh < 0) {
251 perf[p].kh = (dx * pow(kykz, 0.5));
252 }
253 break;
254 }
255 case Y_DIRECTION: {
256 OCP_DBL kzkx = myBulk.rockKz[Idb] * myBulk.rockKx[Idb];
257 OCP_DBL kz_kx = myBulk.rockKz[Idb] / myBulk.rockKx[Idb];
258 assert(kzkx > 0);
259 ro =
260 0.28 *
261 pow((dz * dz * pow(1 / kz_kx, 0.5) + dx * dx * pow(kz_kx, 0.5)),
262 0.5);
263 ro /= (pow(kz_kx, 0.25) + pow(1 / kz_kx, 0.25));
264
265 if (perf[p].kh < 0) {
266 perf[p].kh = (dy * pow(kzkx, 0.5));
267 }
268 break;
269 }
270 case Z_DIRECTION: {
271 OCP_DBL kxky = myBulk.rockKx[Idb] * myBulk.rockKy[Idb];
272 OCP_DBL kx_ky = myBulk.rockKx[Idb] / myBulk.rockKy[Idb];
273 assert(kxky > 0);
274 ro =
275 0.28 *
276 pow((dx * dx * pow(1 / kx_ky, 0.5) + dy * dy * pow(kx_ky, 0.5)),
277 0.5);
278 ro /= (pow(kx_ky, 0.25) + pow(1 / kx_ky, 0.25));
279
280 if (perf[p].kh < 0) {
281 perf[p].kh = (dz * pow(kxky, 0.5));
282 }
283 break;
284 }
285 default:
286 OCP_ABORT("Wrong direction of perforations!");
287 }
288 perf[p].WI = CONV2 * (2 * PI) * perf[p].kh /
289 (log(ro / perf[p].radius) + perf[p].skinFactor);
290 }

589void Well::CalInjdG(const Bulk& myBulk)

597
598 if (depth <= perf.front().depth) {
599 // Well is higher
600 for (OCP_INT p = numPerf - 1; p >= 0; p--) {
601 if (p == 0) {
602 seg_num = ceil(fabs((perf[0].depth - depth) / maxlen));
603 if (seg_num == 0) continue;
604 seg_len = (perf[0].depth - depth) / seg_num;
605 } else {
606 seg_num = ceil(fabs((perf[p].depth - perf[p - 1].depth) / maxlen));
607 if (seg_num == 0) continue;
608 seg_len = (perf[p].depth - perf[p - 1].depth) / seg_num;
609 }
610 OCP_USI n = perf[p].location;

616 for (USI i = 0; i < seg_num; i++) {
617 Ptmp -=
618 myBulk.flashCal[pvtnum]->RhoPhase(Ptmp, myBulk.T, opt.zi.data()) *
619 GRAVITY_FACTOR * seg_len;
620 }
621 dGperf[p] = Pperf - Ptmp;
622 }
623 dG[0] = dGperf[0];
624 for (USI p = 1; p < numPerf; p++) {
625 dG[p] = dG[p - 1] + dGperf[p];

632 if (seg_num == 0) continue;
633 seg_len = (depth - perf[numPerf - 1].depth) / seg_num;
634 } else {
635 seg_num = ceil(fabs((perf[p + 1].depth - perf[p].depth) / maxlen));
636 if (seg_num == 0) continue;
637 seg_len = (perf[p + 1].depth - perf[p].depth) / seg_num;
638 }
639 OCP_USI n = perf[p].location;
640 perf[p].P = BHP + dG[p];

646 Ptmp +=
647 myBulk.flashCal[pvtnum]->RhoPhase(Ptmp, myBulk.T, opt.zi.data()) *

649 }

1205void Well::AssembleMatINJ_IMPEC(const Bulk& myBulk, LinearSystem& myLS,
1206 const OCP_DBL& dt) const
1207{
1208 OCP_FUNCNAME;
1209
1210 const USI nc = myBulk.numCom;
1211 const OCP_USI wId = myLS.dim;
1212 // important !
1213 myLS.dim++;
1214
1215 for (USI p = 0; p < numPerf; p++) {
1216 OCP_USI k = perf[p].location;
1217
1218 OCP_DBL Vfi_zi = 0;
1219 for (USI i = 0; i < nc; i++) {
1220 Vfi_zi += myBulk.vfi[k * nc + i] * opt.zi[i];
1221 }
1222
1223 OCP_DBL valw = dt * perf[p].xi * perf[p].transINJ;
1224 OCP_DBL bw = valw * dG[p];
1225 OCP_DBL valb = valw * Vfi_zi;
1226 OCP_DBL bb = valb * dG[p];
1227
1228 // Bulk to Well
1229
1230 // diag
1231 USI ptr = myLS.diagPtr[k];
1232 myLS.val[k][ptr] += valb;
1233 // off diag
1234 myLS.colId[k].push_back(wId);
1235 myLS.val[k].push_back(-valb);
1236 // b
1237 myLS.b[k] += bb;
1238
1239 // Well to Bulk
1240 switch (opt.optMode) {
1241 case RATE_MODE:
1242 case ORATE_MODE:
1243 case GRATE_MODE:
1244 case WRATE_MODE:
1245 case LRATE_MODE:
1246 // diag
1247 myLS.diagVal[wId] += valw;
1248 // off diag
1249 myLS.colId[wId].push_back(k);
1250 myLS.val[wId].push_back(-valw);
1251 // b
1252 myLS.b[wId] -= bw;
1253 break;
1254 case BHP_MODE:
1255 myLS.colId[wId].push_back(k);
1256 myLS.val[wId].push_back(0);
1257 break;
1258 default:
1259 OCP_ABORT("Wrong well option mode!");
1260 }
1261 }
1262
1263 // Well Self
1264 assert(myLS.val[wId].size() == numPerf);
1265 // the order of perforation is not necessarily in order
1266 switch (opt.optMode) {
1267 case RATE_MODE:
1268 case ORATE_MODE:
1269 case GRATE_MODE:
1270 case WRATE_MODE:
1271 case LRATE_MODE:
1272 // diag
1273 myLS.colId[wId].push_back(wId);
1274 myLS.diagPtr[wId] = numPerf;
1275 myLS.val[wId].push_back(myLS.diagVal[wId]);
1276 // b
1277 myLS.b[wId] += dt * opt.maxRate;
1278 break;
1279 case BHP_MODE:
1280 // diag
1281 myLS.colId[wId].push_back(wId);
1282 myLS.diagPtr[wId] = numPerf;

1904 }
1905 }
