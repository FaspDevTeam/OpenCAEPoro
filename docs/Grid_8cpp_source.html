135OCP_DBL Grid::CalAkdOthogonalGrid(const OCP_USI& bId, const OCP_USI& eId, const USI& direction)
136{
137 OCP_DBL T1;
138 OCP_DBL T2;
139 switch (direction)
140 {
141 case 1:
142 // x-direction
143 T1 = kx[bId] * ntg[bId] * dy[bId] * dz[bId] / dx[bId];
144 T2 = kx[eId] * ntg[eId] * dy[eId] * dz[eId] / dx[eId];
145 return (2 / (1 / T1 + 1 / T2));
146 break;
147 case 2:
148 // y-direction
149 T1 = ky[bId] * ntg[bId] * dz[bId] * dx[bId] / dy[bId];
150 T2 = ky[eId] * ntg[eId] * dz[eId] * dx[eId] / dy[eId];
151 return (2 / (1 / T1 + 1 / T2));
152 break;
153 case 3:
154 // z-direction -- no ntg
155 T1 = kz[bId] * dx[bId] * dy[bId] / dz[bId];
156 T2 = kz[eId] * dx[eId] * dy[eId] / dz[eId];
157 return (2 / (1 / T1 + 1 / T2));
158 break;
159 default:
160 OCP_ABORT("Wrong Direction!");
161 break;
162 }

166void Grid::CalDepthVOthogonalGrid()
167{
168 depth.resize(numGrid, 0);
169 OCP_USI nxny = nx * ny;
170 // 0th layer
171 for (USI j = 0; j < ny; j++) {
172 for (USI i = 0; i < nx; i++) {
173 OCP_USI id = j * nx + i;
174 depth[id] = tops[id] + dz[id] / 2;
175 }
176 }
177 // 1th - (nz-1)th layer
178 for (USI k = 1; k < nz; k++) {
179 OCP_USI knxny = k * nxny;
180 for (USI j = 0; j < ny; j++) {
181 for (USI i = 0; i < nx; i++) {
182 OCP_USI id = knxny + j * nx + i;
183 depth[id] = depth[id - nxny] + dz[id - nxny] / 2 + dz[id] / 2;
184 }
185 }
186 }
187
188 v.resize(numGrid);
189 for (OCP_USI i = 0; i < numGrid; i++) v[i] = dx[i] * dy[i] * dz[i];

233OCP_DBL Grid::CalAkdCornerGrid(const GeneralConnect& conn)
234{
235 OCP_USI bId = conn.begin;
236 OCP_USI eId = conn.end;
237 OCP_DBL bArea = conn.Ad_dd_begin;
238 OCP_DBL eArea = conn.Ad_dd_end;
239 OCP_DBL T1, T2;
240
241 switch (conn.directiontype)
242 {
243 case 1:
244 T1 = ntg[bId] * kx[bId] * bArea;
245 T2 = ntg[eId] * kx[eId] * eArea;
246 return 1 / (1 / T1 + 1 / T2);
247 break;
248 case 2:
249 T1 = ntg[bId] * ky[bId] * bArea;
250 T2 = ntg[eId] * ky[eId] * eArea;
251 return 1 / (1 / T1 + 1 / T2);
252 break;
253 case 3:
254 T1 = kz[bId] * bArea;
255 T2 = kz[eId] * eArea;
256 return 1 / (1 / T1 + 1 / T2);
257 break;
258 default:
259 OCP_ABORT("Wrong Direction Type!");
260 break;
261 }
262}

265void Grid::CalActiveGrid(const OCP_DBL& e1, const OCP_DBL& e2)
266{
267 activeMap_B2G.reserve(numGrid);
268 activeMap_G2B.resize(numGrid);
269 OCP_USI count = 0;
270 for (OCP_USI n = 0; n < numGrid; n++) {
271 if (poro[n] * ntg[n] < e1 || dx[n] * dy[n] * dz[n] < e2) {
272 activeMap_G2B[n] = GB_Pair(false, 0);
273 continue;
274 }
275 activeMap_B2G.push_back(n);
276 activeMap_G2B[n] = GB_Pair(true, count);
277 count++;
278 }
279 activeGridNum = count;
